64 void register_self() noexcept try
65 {
66  using namespace std::chrono;
67  using namespace kerbal::compatibility::chrono_suffix;
68 
69  kerbal::redis::RedisContext regist_self_conn(redis_hostname.c_str(), redis_port, 1500_ms);
70  if (!regist_self_conn) {
71  LOG_FATAL(0, 0, log_fp, "Regist_self context connect failed.");
72  return;
73  }
74  LOG_INFO(0, 0, log_fp, "Regist_self service get context.");
75 
76  static kerbal::redis::Operation opt(regist_self_conn);
77  while (true) {
78  try {
79  const time_t now = time(NULL);
80  const std::string confirm_time = get_ymd_hms_in_local_time_zone(now);
81  opt.hmset("judge_server:" + judge_server_id,
82  "host_name"_cptr, host_name,
83  "ip"_cptr, ip,
84  "user_name"_cptr, user_name,
85  "listening_pid"_cptr, listening_pid,
86  "last_confirm"_cptr, confirm_time);
87  opt.set("judge_server_confirm:" + judge_server_id, confirm_time);
88  opt.expire("judge_server_confirm:" + judge_server_id, 30_s);
89 
90  static kerbal::redis::Set<std::string> s(regist_self_conn, "online_judger");
91  s.insert(judge_server_id);
92 
93  LOG_DEBUG(0, 0, log_fp, "Register self success");
94  std::this_thread::sleep_for(10_s);
95  } catch (const kerbal::redis::RedisException & e) {
96  EXCEPT_FATAL(0, 0, log_fp, "Register self failed.", e);
97  }
98  }
99 } catch (...) {
100  LOG_FATAL(0, 0, log_fp, "Register self failed. Error information: ", UNKNOWN_EXCEPTION_WHAT);
101  exit(2);
102 }

212  LOG_INFO(0, 0, log_fp, "Connecting main REDIS connection...");
213  kerbal::redis::RedisContext mainRedisConn(redis_hostname.c_str(), redis_port, 100_ms);
214  if (!mainRedisConn) {

218  LOG_INFO(0, 0, log_fp, "Main REDIS connection connected!");
219 
220  #ifdef DEBUG

222  LOG_FATAL(0, 0, log_fp, "Error: ", RunnerError::CHDIR_ERROR);
223  exit(-1);

227  try {
231  std::thread regist(register_self);
232  regist.detach();
233  } catch (const std::exception & e) {
234  EXCEPT_FATAL(0, 0, log_fp, "Register self service process fork failed.", e);
235  exit(-3);

238  LOG_INFO(0, 0, log_fp, "updater starting ...");

240  kerbal::redis::List<std::string> update_queue(mainRedisConn, "update_queue");
241 
242  while (loop) {
243 
244  std::string job_item = "-1,-1";
245  int jobType = -1;
246  int sid = -1;
247  try {
248  job_item = update_queue.block_pop_front(0_s);
249  if (JobBase::isExitJob(job_item) == true) {
250  loop = false;
251  LOG_INFO(0, 0, log_fp, "Get exit job.");
252  continue;
253  }
254  std::tie(jobType, sid) = JobBase::parseJobItem(job_item);
255  LOG_DEBUG(jobType, sid, log_fp, "Master get update job: ", job_item);
256  } catch (const std::exception & e) {
257  EXCEPT_FATAL(0, 0, log_fp, "Fail to fetch job.", e, "job_item: ", job_item);
258  continue;
259  } catch (...) {
260  LOG_FATAL(0, 0, log_fp, "Fail to fetch job. Error info: ", "unknown exception", "job_item: ", job_item);
261  continue;
262  }
263 
264  auto start(std::chrono::steady_clock::now());
265  LOG_DEBUG(jobType, sid, log_fp, "Update job start");
266 
267  kerbal::redis::RedisContext redisConn(redis_hostname.c_str(), redis_port, 100_ms);
268  if (!redisConn) {
269  LOG_FATAL(jobType, sid, log_fp, "Redis connection connect failed!");
270  continue;
271  }
272 
273  std::unique_ptr <mysqlpp::Connection> mysqlConn(new mysqlpp::Connection(false));
274  mysqlConn->set_option(new mysqlpp::SetCharsetNameOption("utf8"));
275  if (!mysqlConn->connect(mysql_database.c_str(), mysql_hostname.c_str(), mysql_username.c_str(), mysql_passwd.c_str())) {
276  LOG_FATAL(jobType, sid, log_fp, "Mysql connection connect failed!");
277  continue;
278  }
279 
280  std::unique_ptr <UpdateJobBase> job = nullptr;
281  try {
282  job = make_update_job(jobType, sid, redisConn, std::move(mysqlConn));
283  } catch (const std::exception & e) {
284  EXCEPT_FATAL(jobType, sid, log_fp, "Job construct failed!", e);
285  continue;
286  } catch (...) {
287  LOG_FATAL(jobType, sid, log_fp, "Job construct failed! Error information: ", UNKNOWN_EXCEPTION_WHAT);
288  continue;
289  }
290 
291  try {
292  job->handle();
293  } catch (const std::exception & e) {
294  EXCEPT_FATAL(jobType, sid, log_fp, "Job handle failed!", e);
295  continue;
296  } catch (...) {
297  LOG_FATAL(jobType, sid, log_fp, "Job handle failed! Error information: ", UNKNOWN_EXCEPTION_WHAT);
298  continue;
299  }
300 
301  auto end(std::chrono::steady_clock::now());
302  auto time_consume = std::chrono::duration_cast<std::chrono::milliseconds>(end - start);
303  LOG_DEBUG(jobType, sid, log_fp, "Update consume: ", time_consume.count());
304  }
305  return 0;
306 } catch (const std::exception & e) {
307  EXCEPT_FATAL(0, 0, log_fp, "An uncaught exception caught by main.", e);
308  throw;
309 } catch (...) {
310  LOG_FATAL(0, 0, log_fp, "An uncaught exception caught by main. Error information: ", UNKNOWN_EXCEPTION_WHAT);
311  throw;
312 }
