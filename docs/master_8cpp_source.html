315 void listenning_loop()
316 {
317  // 连接 redis 的 update_queue 队列，master 据此进行更新操作
318  auto redis_conn_handler = sync_fetch_redis_conn();
319  kerbal::redis_v2::list update_queue(*redis_conn_handler, "update_queue");
320 
321  std::deque<std::future<void> > future_group;
322 
323  while (loop) {
324 
325  int jobType(0);
326  ojv4::s_id_type s_id(0);
327 
328  std::exception_ptr queue_pop_exception = nullptr;
329  std::exception_ptr job_parse_exception = nullptr;
330 
331  /*
332  * 当收到 SIGTERM 信号时，会在评测队列末端加一个特殊的评测任务用于标识停止测评。此处若检测到停止
333  * 工作的信号，则结束工作loop
334  * 若取得的是正常的评测任务则继续工作
335  */
336  std::string job_item;
337  try {
338  job_item = update_queue.blpop(0_s);
339  } catch (const std::exception & e) {
340  EXCEPT_FATAL(0, 0, log_fp, "Fail to fetch job.", e);
341  queue_pop_exception = std::current_exception();
342  //DO NOT THROW
343  }
344 
345  try {
346  std::tie(jobType, s_id) = JobBase::parseJobItem(job_item);
347  } catch (const std::exception & e) {
348  EXCEPT_FATAL(0, 0, log_fp, "Fail to parse job item.", e, "job_item: ", job_item);
349  continue;
350  }
351 
352  std::unique_ptr <UpdateJobBase> job = nullptr;
353  try {
354  // 根据 jobType 与 sid 获取 UpdateJobBase 通用类型的 update job 信息
355  // 信息从 redis 数据库读入，并且将一个 mysql 数据库的连接控制权转移至实例化的 job 当中
356  // 在实例化之前就建立好 mysql 连接再转移给它，避免了进入更新操作内部之后才发现无法连接 mysql 数据库
357  // 避免了无效操作的消耗与可能导致的其他逻辑混乱
358  // 此外，unique_ptr 的特性决定了其不应该使用值传递的方式。
359  // 而使用 std::move 将其强制转换为右值引用，是为了使用移动构造函数来优化性能（吗？这个不确定）
360  auto redis_conn_handler = sync_fetch_redis_conn();
361  job = make_update_job(jobType, s_id, *redis_conn_handler);
362  } catch (const std::exception & e) {
363  EXCEPT_FATAL(jobType, s_id, log_fp, "Job construct failed!", e);
364  continue;
365  } catch (...) {
366  UNKNOWN_EXCEPT_FATAL(jobType, s_id, log_fp, "Job construct failed!");
367  continue;
368  }
369 
370  try {
371  // 执行本次 update job
372  future_group.push_back(std::async(std::launch::async,
373  [j=std::move(job), jobType, s_id]() {
374  // 本次更新开始时间
375  PROFILE_HEAD
376  try {
377  j->handle();
378  } catch (const std::exception & e) {
379  EXCEPT_FATAL(jobType, s_id, log_fp, "Job handle failed!", e);
380  throw e;
381  }
382  // 本次更新结束时间
383  PROFILE_TAIL(jobType, s_id, log_fp, "Update finished");
384  LOG_INFO(jobType, s_id, log_fp, "Update finished");
385  }
386  ));
387  } catch (const std::exception & e) {
388  EXCEPT_FATAL(jobType, s_id, log_fp, "Job handle failed!", e);
389  } catch (...) {
390  UNKNOWN_EXCEPT_FATAL(jobType, s_id, log_fp, "Job handle failed!");
391  }
392 
393  auto update_queue_is_empty = [&update_queue]() noexcept {
394  try {
395  return update_queue.llen() == 0;
396  } catch(...) {
397  return true;
398  }
399  };
400 
401  while (future_group.size() >= 25) {
402  try {
403  future_group.front().get();
404  future_group.pop_front();
405  } catch (const std::exception & e) {
406  EXCEPT_FATAL(0, 0, log_fp, "Future get failed!", e);
407  }
408  }
409 
410  while (update_queue_is_empty() && !future_group.empty()) {
411  try {
412  future_group.front().get();
413  future_group.pop_front();
414  } catch (const std::exception & e) {
415  EXCEPT_FATAL(0, 0, log_fp, "Future get failed!", e);
416  }
417  }
418  }
419 
420  while (!future_group.empty()) {
421  try {
422  future_group.front().get();
423  future_group.pop_front();
424  } catch (const std::exception & e) {
425  EXCEPT_FATAL(0, 0, log_fp, "Future get failed!", e);
426  }
427  }
428 }
