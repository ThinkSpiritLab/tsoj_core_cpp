24 std::pair<int, int> JobBase::parseJobItem(const std::string & args)
25 {
26  std::string job_item = args;
27  std::string::size_type cut_point = job_item.find(',');
28 
29  if (cut_point == std::string::npos) {
30  std::invalid_argument e("Invalid job_item arguments: " + job_item);
31  EXCEPT_FATAL(0, 0, log_fp, "Parse job item failed.", e);
32  throw e;
33  }
34 
35  job_item[cut_point] = '\0';
36 
37  try {
38  int job_type = std::stoi(job_item.c_str());
39  int job_id = std::stoi(job_item.c_str() + cut_point + 1);
40  return std::make_pair(job_type, job_id);
41  } catch (const std::invalid_argument & ) {
42  std::invalid_argument e("Invalid job_item arguments: " + job_item);
43  EXCEPT_FATAL(0, 0, log_fp, "Parse job item failed.", e);
44  throw e;
45  }
46 }

48 JobBase::JobBase(int jobType, int sid, const kerbal::redis::RedisContext & redisConn) :
49  jobType(jobType), sid(sid), redisConn(redisConn)
50 {
51  using std::chrono::milliseconds;
52  using kerbal::utility::MB;
53  static boost::format key_name_tmpl("job_info:%d:%d");
54  const std::string job_info_key = (key_name_tmpl % jobType % sid).str();
55 
56  kerbal::redis::Operation opt(redisConn);
57 
58  try {
59  this->pid = opt.hget<int>(job_info_key, "pid");
60  this->lang = (Language) (opt.hget<int>(job_info_key, "lang"));
61  this->cases = opt.hget<int>(job_info_key, "cases");
62  this->timeLimit = milliseconds(opt.hget<int>(job_info_key, "time_limit"));
63  this->memoryLimit = MB(opt.hget<unsigned long long>(job_info_key, "memory_limit"));
64  this->similarity_threshold = opt.hget<int>(job_info_key, "sim_threshold");
65  } catch (const RedisNilException & e) {
66  EXCEPT_FATAL(jobType, sid, log_fp, "Job details lost.", e);
67  throw;
68  } catch (const RedisUnexpectedCaseException & e) {
69  EXCEPT_FATAL(jobType, sid, log_fp, "Redis returns an unexpected type.", e);
70  throw;
71  } catch (const RedisException & e) {
72  EXCEPT_FATAL(jobType, sid, log_fp, "Fail to fetch job details.", e);
73  throw;
74  } catch (const std::exception & e) {
75  EXCEPT_FATAL(jobType, sid, log_fp, "Fail to fetch job details.", e);
76  throw;
77  }
78 }

80 kerbal::redis::RedisReply JobBase::get_source_code() const
81 {
82  static RedisCommand get_src_code_templ("hget source_code:%d:%d source");
83  RedisReply reply;
84  try {
85  reply = get_src_code_templ.execute(redisConn, this->jobType, this->sid);
86  } catch (const std::exception & e) {
87  EXCEPT_FATAL(jobType, sid, log_fp, "Get source code failed!", e);
88  throw;
89  }
90  if (reply.replyType() != RedisReplyType::STRING) {
91  RedisUnexpectedCaseException e(reply.replyType());
92  EXCEPT_FATAL(jobType, sid, log_fp, "Get source code failed!", e);
93  throw e;
94  }
95  return reply;
96 }

98 void JobBase::storeSourceCode() const
99 {
100  RedisReply reply = this->get_source_code();
101 
102  static constexpr const char * stored_file_name[] = {"Main.c", "Main.cpp", "Main.java"};
103  size_t i = 1;
104  switch (lang) {
105  case Language::Cpp:
106  case Language::Cpp14:
107  i = 1;
108  break;
109  case Language::C:
110  i = 0;
111  break;
112  case Language::Java:
113  i = 2;
114  break;
115  }
116  std::ofstream fout(stored_file_name[i], std::ios::out);
117  if (!fout) {
118  LOG_FATAL(jobType, sid, log_fp, "Open source code file failed");
119  throw JobHandleException("Open source code file failed");
120  }
121  fout << reply->str;
122  if (fout.bad()) {
123  LOG_FATAL(jobType, sid, log_fp, "Store source code failed");
124  throw JobHandleException("Store source code failed");
125  }
126 }
