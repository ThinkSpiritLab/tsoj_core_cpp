28 std::pair<int, ojv4::s_id_type> JobBase::parseJobItem(const std::string & args)
29 {
30  std::string job_item = args;
31  std::string::size_type cut_point = job_item.find(',');
32 
33  if (cut_point == std::string::npos) {
34  std::invalid_argument e("Invalid job_item arguments: " + job_item);
35  EXCEPT_FATAL(0, 0, log_fp, "Parse job item failed.", e);
36  throw e;
37  }
38 
39  job_item[cut_point] = '\0';
40 
41  try {
42  int job_type = std::stoi(job_item.c_str());
43  ojv4::s_id_type job_id(std::stoi(job_item.c_str() + cut_point + 1));
44  return std::make_pair(job_type, job_id);
45  } catch (const std::invalid_argument & ) {
46  std::invalid_argument e("Invalid job_item arguments: " + job_item);
47  EXCEPT_FATAL(0, 0, log_fp, "Parse job item failed.", e);
48  throw e;
49  }
50 }

52 JobBase::JobBase(int jobType, ojv4::s_id_type s_id, kerbal::redis_v2::connection & redis_conn) :
53  jobType(jobType), s_id(s_id)
54 {
55  try {
56  using boost::lexical_cast;
57  using namespace kerbal::utility;
58  namespace optional_ns = kerbal::data_struct;
59  using namespace optional_ns;
60 
61  kerbal::redis_v2::hash h(redis_conn, "job_info:%d:%d"_fmt(jobType, s_id).str());
62  std::vector<optional<std::string>> res = h.hmget(
63  "pid",
64  "lang",
65  "cases",
66  "time_limit",
67  "memory_limit",
68  "sim_threshold");
69 
70  if (res[0].empty())
71  throw std::runtime_error("lack pid");
72  this->p_id = lexical_cast<ojv4::p_id_type>(res[0].ignored_get());
73 
74  if (res[1].empty())
75  throw std::runtime_error("lack lang");
76  this->lang = (Language) (lexical_cast<int>(res[1].ignored_get()));
77 
78  this->cases = res[2].empty() ? 1 : lexical_cast<int>(res[2].ignored_get());
79 
80  ojv4::s_time_literal time_limit_literal = res[3].empty() ? 0 : lexical_cast<ojv4::s_time_literal>(res[3].ignored_get());
81  this->time_limit = ojv4::s_time_type<std::milli>(time_limit_literal);
82 
83  ojv4::s_mem_literal mem_limit_literal = res[4].empty() ? 0 : lexical_cast<ojv4::s_mem_literal>(res[4].ignored_get());
84  this->memory_limit = ojv4::s_mem_type<kerbal::utility::mebi>(mem_limit_literal);
85 
86  this->similarity_threshold = lexical_cast<ojv4::s_similarity_type>(res[5].empty() ? 0 : res[5].ignored_get());
87 
88  } catch (const std::exception & e) {
89  EXCEPT_FATAL(jobType, s_id, log_fp, "Fail to fetch job details.", e);
90  throw;
91  }
92 }

94 kerbal::redis_v2::reply JobBase::get_source_code() const
95 {
96  using namespace kerbal::redis;
97 
98  kerbal::redis_v2::reply reply;
99  std::vector<std::string> argv = { "hget", "source_code:%d:%d"_fmt(jobType, s_id).str(), "source" };
100  try {
101  auto redis_conn_handler = sync_fetch_redis_conn();
102  reply = redis_conn_handler->argv_execute(argv.begin(), argv.end());
103  } catch (const std::exception & e) {
104  EXCEPT_FATAL(jobType, s_id, log_fp, "Get source code failed!", e);
105  throw;
106  }
107  if (reply.type() != kerbal::redis_v2::reply_type::STRING) {
108  kerbal::redis_v2::unexpected_case_exception e(reply.type(), argv.begin(), argv.end());
109  EXCEPT_FATAL(jobType, s_id, log_fp, "Get source code failed!", e);
110  throw e;
111  }
112  return reply;
113 }

115 void JobBase::storeSourceCode(const std::string & parent_path_args, const std::string & file_name) const
116 {
117  using namespace kerbal::redis;
118 
119  kerbal::redis_v2::reply reply = this->get_source_code();
120 
121  std::string parent_path = parent_path_args;
122  while(!parent_path.empty() && std::isblank(parent_path.back())) {
123  parent_path.pop_back();
124  }
125  if (parent_path.size() == 0) {
126  parent_path = "./";
127  } else {
128  if (parent_path.back() != '/') {
129  parent_path += '/';
130  }
131  mkdir_p(parent_path);
132  }
133 
134  std::ofstream fout(parent_path + file_name + '.' + source_file_suffix(lang), std::ios::out);
135  if (!fout) {
136  LOG_FATAL(jobType, s_id, log_fp, "Open source code file failed");
137  throw JobHandleException("Open source code file failed");
138  }
139  fout << reply->str;
140  if (fout.bad()) {
141  LOG_FATAL(jobType, s_id, log_fp, "Store source code failed");
142  throw JobHandleException("Store source code failed");
143  }
144 }
