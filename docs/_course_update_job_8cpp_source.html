40 void CourseUpdateJob::update_problem_submit_and_accept_num_in_this_course()
41 {
42  mysqlpp::Query query = mysqlConn->query(
43  "select u_id, s_result from solution where p_id = %0 and c_id = %1"
44  );
45  query.parse();
46 
47  std::set<int> accepted_user;
48  int submit_num = 0;
49  {
50  mysqlpp::StoreQueryResult res = query.store(this->pid, this->cid);
51 
52  for (const auto & row : res) {
53  int u_id_in_this_row = row["u_id"];
54  // 此题已通过的用户的集合中无此条 solution 对应的用户
55  if (accepted_user.find(u_id_in_this_row) == accepted_user.end()) {
56  UnitedJudgeResult result = UnitedJudgeResult(int(row["s_result"]));
57  if (result == UnitedJudgeResult::ACCEPTED) {
58  accepted_user.insert(u_id_in_this_row);
59  }
60  ++submit_num;
61  }
62  }
63 
64  if (query.errnum() != 0) {
65  MysqlEmptyResException e(query.errnum(), query.error());
66  EXCEPT_FATAL(jobType, sid, log_fp, "Query problem's solutions failed!", e);
67  throw e;
68  }
69  }
70 
71  mysqlpp::Query update = mysqlConn->query(
72  "update course_problem set c_p_submit = %0, c_p_accept = %1 where p_id = %2 and c_id = %3"
73  );
74  update.parse();
75 
76  mysqlpp::SimpleResult res = update.execute(submit_num, accepted_user.size(), this->pid, this->cid);
77  if (!res) {
78  MysqlEmptyResException e(query.errnum(), query.error());
79  EXCEPT_FATAL(jobType, sid, log_fp, "Update problem's submit and accept number in this course failed!", e, ", course id: ", this->cid);
80  throw e;
81  }
82 }

183 void CourseUpdateJob::update_user_problem_status()
184 {
185  LOG_DEBUG(jobType, sid, log_fp, "CourseUpdateJob::update_user_problem");
186 
187  // 先更新练习视角的用户通过情况表
188  // 使课程中（cid 非空） user_problem 中某个 user 对某题的状态同步到非课程的 user_problem 中（cid 为空）
189  this->supper_t::update_user_problem_status();
190 
191  user_problem_status old_status = this->get_course_user_problem_status();
192 
193  bool is_ac = this->result.judge_result == UnitedJudgeResult::ACCEPTED ;
194 
195  switch(old_status) {
196  case user_problem_status::TODO: {
197  mysqlpp::Query insert = mysqlConn->query("insert into user_problem (u_id, p_id, c_id, status) "
198  "values (%0, %1, %2, %3)");
199  insert.parse();
200  mysqlpp::SimpleResult ret = insert.execute(uid, pid, cid, is_ac ? 0 : 1);
201  if (!ret) {
202  MysqlEmptyResException e(insert.errnum(), insert.error());
203  EXCEPT_FATAL(jobType, sid, log_fp, "Update user_problem failed!", e);
204  throw e;
205  }
206  break;
207  }
208  case user_problem_status::ATTEMPTED: {
209  mysqlpp::Query update = mysqlConn->query(
210  "update user_problem set status = %0 "
211  "where u_id = %1 and p_id = %2 and c_id = %3"
212  );
213  update.parse();
214  mysqlpp::SimpleResult ret = update.execute(is_ac ? 0 : 1, uid, pid, cid);
215  if (!ret) {
216  MysqlEmptyResException e(update.errnum(), update.error());
217  EXCEPT_FATAL(jobType, sid, log_fp, "Update user_problem failed!", e);
218  throw e;
219  }
220  break;
221  }
222  case user_problem_status::ACCEPTED:
223  return;
224  default:
225  LOG_FATAL(jobType, sid, log_fp, "Undefined user's problem status!");
226  throw std::logic_error("Undefined user's problem status!");
227  }
228 }
