98  template <typename ChildInstanceType = InstanceType, typename ... ConstructArgs>
99  static std::pair<size_t, std::exception_ptr> construct(size_t num, ConstructArgs&& ... args) noexcept
100  {
101  std::lock_guard<std::mutex> lck (get_mutex());
102  for(size_t i = 0; i < num; ++i) {
103  try {
104  get_pool().push_back(new ChildInstanceType(std::forward<ConstructArgs>(args)...));
105  } catch(...) {
106  return std::make_pair(i, std::current_exception());
107  }
108  }

112  static auto_revert_t fetch()
113  {

115  while(!get_pool().empty()) {
116  InstanceType* p = get_pool().front();
117  get_pool().pop_front();
