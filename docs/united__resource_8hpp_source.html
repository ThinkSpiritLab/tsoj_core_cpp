26 inline const char* getJudgeStatusName(JudgeStatus status)
27 {

32  return "COMPILING";

34  return "RUNNING";
35  case JudgeStatus::JUDGING :
36  return "JUDGING";
37  case JudgeStatus::FINISHED :

40  return "UPDATED";
41  }
42  return "Unknown JudgeStatus";
43 }
