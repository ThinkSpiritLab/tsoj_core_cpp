24 make_update_job(int jobType, int sid, const kerbal::redis::RedisContext & redisConn,
25  std::unique_ptr<mysqlpp::Connection> && mysqlConn)
26 {
27  UpdateJobBase * job = nullptr;
28 
29  try {
30  if (jobType == 0) {
31  int cid = 0;
32  static boost::format key_name_tmpl("job_info:0:%d");
33  kerbal::redis::Operation opt(redisConn);
34 
35  try {
36  cid = opt.hget<int>((key_name_tmpl % sid).str(), "cid");
37  } catch (const std::exception & e) {
38  EXCEPT_FATAL(jobType, sid, log_fp, "Get cid failed!", e);
39  throw;
40  } catch (...) {
41  LOG_FATAL(jobType, sid, log_fp, "Get cid failed! Error information: ", UNKNOWN_EXCEPTION_WHAT);
42  throw;
43  }
44  if (cid != 0) {
45  job = new CourseUpdateJob(jobType, sid, redisConn, std::move(mysqlConn));
46  } else {
47  job = new ExerciseUpdateJob(jobType, sid, redisConn, std::move(mysqlConn));
48  }
49  } else {
50  job = new ContestUpdateJob(jobType, sid, redisConn, std::move(mysqlConn));
51  }
52  } catch (const std::exception & e) {
53  EXCEPT_FATAL(jobType, sid, log_fp, "Error occurred while construct update job!", e);
54  throw;
55  } catch (...) {
56  LOG_FATAL(jobType, sid, log_fp, "Error occurred while construct update job! Error information: ", UNKNOWN_EXCEPTION_WHAT);
57  throw;
58  }
59  return std::unique_ptr<UpdateJobBase>(job);
60 }

62 UpdateJobBase::UpdateJobBase(int jobType, int sid, const RedisContext & redisConn,
63  std::unique_ptr<mysqlpp::Connection> && mysqlConn) :
64  supper_t(jobType, sid, redisConn), mysqlConn(std::move(mysqlConn))
65 {
66  static boost::format key_name_tmpl("job_info:%d:%d");
67  const std::string key_name = (key_name_tmpl % jobType % sid).str();
68 
69  kerbal::redis::Operation opt(redisConn);
70 
71  // 取 job 信息
72  try {
73  this->uid = opt.hget<int>(key_name, "uid");
74  this->cid = opt.hget<int>(key_name, "cid");
75  this->postTime = opt.hget(key_name, "post_time");
76  //this->haveAccepted = (bool) opt.hget<int>(key_name, "have_accepted");
77  this->no_store_ac_code = (bool) opt.hget<int>(key_name, "no_store_ac_code");
78  this->is_rejudge = (bool) opt.hget<int>(key_name, "is_rejudge");
79  } catch (const RedisNilException & e) {
80  EXCEPT_FATAL(jobType, sid, log_fp, "job details lost.", e);
81  throw;
82  } catch (const RedisUnexpectedCaseException & e) {
83  EXCEPT_FATAL(jobType, sid, log_fp, "redis returns an unexpected type.", e);
84  throw;
85  } catch (const RedisException & e) {
86  EXCEPT_FATAL(jobType, sid, log_fp, "Fail to fetch job details.", e);
87  throw;
88  } catch (const std::exception & e) {
89  EXCEPT_FATAL(jobType, sid, log_fp, "Fail to fetch job details.", e);
90  throw;
91  }
92 
93  static boost::format judge_status_name_tmpl("judge_status:%d:%d");
94  const std::string judge_status = (judge_status_name_tmpl % jobType % sid).str();
95 
96  // 取评测结果
97  try {
98  this->result.judge_result = (UnitedJudgeResult) opt.hget<int>(judge_status, "result");
99  this->result.cpu_time = std::chrono::milliseconds(opt.hget<int>(judge_status, "cpu_time"));
100  this->result.real_time = std::chrono::milliseconds(opt.hget<int>(judge_status, "real_time"));
101  this->result.memory = kerbal::utility::Byte(opt.hget<int>(judge_status, "memory"));
102  this->result.similarity_percentage = opt.hget<int>(judge_status, "similarity_percentage");
103  } catch (const RedisNilException & e) {
104  EXCEPT_FATAL(jobType, sid, log_fp, "job details lost.", e);
105  throw;
106  } catch (const RedisUnexpectedCaseException & e) {
107  EXCEPT_FATAL(jobType, sid, log_fp, "redis returns an unexpected type.", e);
108  throw;
109  } catch (const RedisException & e) {
110  EXCEPT_FATAL(jobType, sid, log_fp, "Fail to fetch job details.", e);
111  throw;
112  } catch (const std::exception & e) {
113  EXCEPT_FATAL(jobType, sid, log_fp, "Fail to fetch job details.", e);
114  throw;
115  }

118 void UpdateJobBase::handle()
119 {
120  std::exception_ptr update_solution_exception;
121  try {
122  this->update_solution();
123  } catch (const std::exception & e) {
124  update_solution_exception = std::current_exception();
125  EXCEPT_FATAL(jobType, sid, log_fp, "Update solution failed!", e);
126  //DO NOT THROW
127  }
128 
129  std::exception_ptr update_user_and_problem_exception;
130  if(update_solution_exception == nullptr) {
131  //如果 update solution 失败, 则跳过 update user and problem
132  //注意!!! 更新用户提交数通过数, 题目提交数通过数的操作必须在更新用户题目完成状态成功之前
133  try {
134  this->update_user_and_problem();
135  } catch (const std::exception & e) {
136  update_user_and_problem_exception = std::current_exception();
137  EXCEPT_FATAL(jobType, sid, log_fp, "Update user and problem failed!", e);
138  //DO NOT THROW
139  }
140  } else {
141  LOG_WARNING(jobType, sid, log_fp, "Escape update user and problem because update solution failed!");
142  }
143 
144  std::exception_ptr update_user_problem_exception;
145  if(update_solution_exception == nullptr) { //如果 update solution 失败, 则跳过 update user problem
146  try {
147  this->update_user_problem();
148  } catch (const std::exception & e) {
149  update_user_problem_exception = std::current_exception();
150  EXCEPT_FATAL(jobType, sid, log_fp, "Update user problem failed!", e);
151  //DO NOT THROW
152  }
153  } else {
154  LOG_WARNING(jobType, sid, log_fp, "Escape update user problem because update solution failed!");
155  }
156 
157  std::exception_ptr update_source_code_exception;
158  if(update_solution_exception == nullptr) { //如果 update solution 失败, 则跳过 update source code
159  try
160  { // update source code
161  RedisReply reply = this->get_source_code();
162  this->update_source_code(reply->str);
163  } catch (const std::exception & e) {
164  update_source_code_exception = std::current_exception();
165  EXCEPT_WARNING(jobType, sid, log_fp, "Update source code failed!", e);
166  //DO NOT THROW
167  }
168  } else {
169  LOG_WARNING(jobType, sid, log_fp, "Escape update source code because update solution failed!");
170  }
171 
172  std::exception_ptr update_compile_info_exception;
173  //保存编译信息
174  if (result.judge_result == UnitedJudgeResult::COMPILE_ERROR) {
175  if(update_solution_exception == nullptr) { //如果 update solution 失败, 则跳过 update compile info
176  try {
177  RedisReply reply = this->get_compile_info();
178  this->update_compile_info(reply->str);
179  } catch (const std::exception & e) {
180  update_compile_info_exception = std::current_exception();
181  EXCEPT_WARNING(jobType, sid, log_fp, "Update compile info failed!", e);
182  //DO NOT THROW
183  }
184  } else {
185  LOG_WARNING(jobType, sid, log_fp, "Escape update compile info because update solution failed!");
186  }
187  }
188 
189  Operation opt(redisConn);
190 
191  if (update_solution_exception == nullptr
192  && update_user_and_problem_exception == nullptr
193  && update_user_problem_exception == nullptr
194  && update_source_code_exception == nullptr
195  && update_compile_info_exception == nullptr) {
196  this->commitJudgeStatusToRedis(JudgeStatus::UPDATED);
197 
198  this->clear_this_jobs_info_in_redis();
199 
200  } else {
201  LOG_WARNING(jobType, sid, log_fp, "Error occurred while update this job!");
202  }
203  this->clear_previous_jobs_info_in_redis();
204 }

224 void UpdateJobBase::clear_previous_jobs_info_in_redis() noexcept try
225 {
226  // del solution - 600
227 
228  if (sid > 600) {
229  int prev_sid = this->sid - 600;
230 
231  Operation opt(this->redisConn);
232 
233  static boost::format judge_status_templ("judge_status:%d:%d");
234 
235  JudgeStatus judge_status = JudgeStatus::UPDATED;
236  try {
237  judge_status = (JudgeStatus) opt.hget<int>((judge_status_templ % jobType % prev_sid).str(), "status");
238  } catch (const std::exception & e) {
239  EXCEPT_FATAL(jobType, sid, log_fp, "Get judge status failed!", e);
240  return; //DO NOT THROW
241  }
242 
243  if (judge_status == JudgeStatus::UPDATED) {
244  try {
245  if (opt.del((judge_status_templ % jobType % prev_sid).str()) == false) {
246  LOG_WARNING(jobType, sid, log_fp, "Doesn't delete judge_status actually!");
247  }
248  } catch (const std::exception & e) {
249  LOG_WARNING(jobType, sid, log_fp, "Exception occurred while deleting judge_status!");
250  //DO NOT THROW
251  }
252 
253  try {
254  static boost::format job_info("job_info:%d:%d");
255  if (opt.del((job_info % jobType % prev_sid).str()) == false) {
256  LOG_WARNING(jobType, sid, log_fp, "Doesn't delete job_info actually!");
257  }
258  } catch (const std::exception & e) {
259  LOG_WARNING(jobType, sid, log_fp, "Exception occurred while deleting job_info!");
260  //DO NOT THROW
261  }
262  }
263  }
264 } catch(...) {
265  LOG_FATAL(jobType, sid, log_fp, "Clear previous jobs info in redis failed! Error information: ", UNKNOWN_EXCEPTION_WHAT);
266 }
