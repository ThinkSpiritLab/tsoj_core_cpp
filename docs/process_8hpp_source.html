41  template <typename Callable, typename ... Args>
42  explicit process(Callable && func, Args && ... args) :
43  father_id(getpid()), child_id(-1), status(joined)
44  {
45  child_id = fork();
46  if (child_id == -1) {
47  status = none;
48  throw std::exception();
49  } else if (child_id == 0) {
50  {
51  func(std::forward<Args>(args)...);
52  }
53  exit(0);
54  }
