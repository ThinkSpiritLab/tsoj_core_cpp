38  void add_solution(ojv4::u_id_type u_id, ojv4::s_result_enum s_result)
39  {
40  switch (s_result) {
41  case ojv4::s_result_enum::ACCEPTED:
42  if (accepted_users.insert(u_id).second == true) {
43  // 当此题已通过的用户的集合中无此条 solution 对应的用户时, insert 操作返回 true
44  ++__submit_num;
45  }
46  break;
47  case ojv4::s_result_enum::SYSTEM_ERROR: // ignore system error
48  break;
49  default:
50  if (accepted_users.find(u_id) == accepted_users.end())
51  ++__submit_num;
52  break;
53  }
54  }
