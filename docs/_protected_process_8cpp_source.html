44  void __dup(const ProtectedProcessConfig & config)
45  {
46  {
47  FILE * input_file = fopen(config.input_path.c_str(), "r");
48  file_guard input_file_guard(input_file);
49 
50  if (input_file == nullptr) {
51  throw Dup2FailedExcetion(Dup2FailedExcetion::DupFileType::INPUT_FILE, config.input_path);
52  }
53  // redirect file -> stdin
54  // On success, these system calls return the new descriptor.
55  // On error, -1 is returned, and errno is set appropriately.
56  if (dup2(fileno(input_file), fileno(stdin)) == -1) {
57  throw Dup2FailedExcetion(Dup2FailedExcetion::DupFileType::INPUT_FILE, config.input_path);
58  }
59  }
60 
61  {
62  FILE * output_file = fopen(config.output_path.c_str(), "w");
63  file_guard output_file_guard(output_file);
64 
65  if (output_file == nullptr) {
66  throw Dup2FailedExcetion(Dup2FailedExcetion::DupFileType::OUTPUT_FILE, config.output_path);
67  }
68  if (dup2(fileno(output_file), fileno(stdout)) == -1) {
69  throw Dup2FailedExcetion(Dup2FailedExcetion::DupFileType::OUTPUT_FILE, config.output_path);
70  }
71  }
72 
73  {
74  FILE * error_file = fopen(config.error_path.c_str(), "w");
75  file_guard error_file_guard(error_file);
76 
77  if (error_file == nullptr) {

87  void __set_rlimit(const ProtectedProcessConfig & config)
88  {
89  using namespace std::chrono;
90  using namespace kerbal::compatibility::chrono_suffix;
91  using namespace kerbal::utility;
92 
93  if (config.max_stack) {
94  rlim_t rlim = static_cast<rlim_t>(storage_cast<Byte>(config.max_stack.value()).count());
95  struct rlimit max_stack = {
96  .rlim_cur = rlim,
97  .rlim_max = rlim,
98  };
99  if (setrlimit(RLIMIT_STACK, &max_stack) != 0) {
100  throw SetRlimitFailedException();
101  }
102  }
103 
104  // set memory limit
105  if (config.max_memory) {
106  rlim_t rlim = static_cast<rlim_t>(storage_cast<Byte>(config.max_memory.value()).count() * 2);
107  struct rlimit max_memory = {
108  .rlim_cur = rlim,
109  .rlim_max = rlim,
110  };
111  if (setrlimit(RLIMIT_AS, &max_memory) != 0) {
112  throw SetRlimitFailedException();
113  }
114  }
115 
116  // set cpu time limit (in seconds)
117  if (config.max_cpu_time) {
118  rlim_t rlim = static_cast<rlim_t>(duration_cast<seconds>(config.max_cpu_time.value() + 1000_ms).count());
119  struct rlimit max_cpu_time = {
120  .rlim_cur = rlim,
121  .rlim_max = rlim,
122  };
123  if (setrlimit(RLIMIT_CPU, &max_cpu_time) != 0) {
124  throw SetRlimitFailedException();
125  }
126  }
127 
128 
129  // set max process number limit

131  rlim_t rlim = static_cast<rlim_t>(config.max_process_number.value());
132  struct rlimit max_process_number = {
133  .rlim_cur = rlim,

135  };
136  if (setrlimit(RLIMIT_NPROC, &max_process_number) != 0) {
137  throw SetRlimitFailedException();

141

157 ProtectedProcessDetails
158 protected_process(const ExecuteArgs & execute_args, const ProtectedProcessConfig & config, const ExecuteArgs & env)
159 {
160  process child_process;
161  // 此处创建了一个运行子进程, 该进程内先加载保护策略, 然后使用 execve 函数用 execute_agrs 替换自身
162  try {
163  child_process = process([&execute_args, &config, &env]() noexcept {
164 
165  std::vector<int> v(4000000);
166  std::cout << v[4] << std::endl;
167 
168  try {
169  try {
170  __set_rlimit(config);
171  } catch(const SetRlimitFailedException & e) {
172  raise(SIGUSR1);
173  return;
174  }
175 
176  try {
177  __dup(config);
178  } catch(const Dup2FailedExcetion & e) {
179  raise(SIGUSR1);
180  return;
181  }
182  } catch(...) {
183  raise(SIGUSR1);
184  return;
185  }
186  execve(execute_args.getArgs()[0], execute_args.getArgs().get(), env.getArgs().get());
187  });
188  } catch (const std::exception & e) {
189  throw std::runtime_error("fork failed!");
190  }
191 
192  using namespace std::chrono;
193  using namespace kerbal::compatibility::chrono_suffix;
194 
195  // record current time
196  auto process_start_time_point = high_resolution_clock::now();
197 
198  if (config.max_real_time.has_value()) {
199  std::thread timeout_killer_thread;
200  try {
201  timeout_killer_thread = std::thread([&child_process](milliseconds timeout) {
202  std::this_thread::sleep_for(timeout + 1_s);
203  child_process.kill(SIGKILL);
204  }, config.max_real_time.value());
205  } catch (const std::system_error & e) {
206  child_process.kill(SIGKILL);
207  throw ThreadFailedException();
208  }
209 
210  try {
211  timeout_killer_thread.detach();
212  } catch (const std::system_error & e) {
213  child_process.kill(SIGKILL);
214  throw ThreadFailedException();
215  }
216  }
217 
218  ProtectedProcessResult running_result = ProtectedProcessResult::ACCEPTED;
219  std::chrono::milliseconds cpu_time;
220  std::chrono::milliseconds real_time;
221  kerbal::utility::KB memory;
222  int exit_code;
223  int status;
224  {
225  struct rusage resource_usage;
226 
227  // wait for child process to terminate
228  // on success, returns the process ID of the child whose state has changed;
229  // On error, -1 is returned.
230  if (child_process.join(&status, WSTOPPED, &resource_usage) == -1) {
231  child_process.kill(SIGKILL);
232  throw std::runtime_error("wait failed!");
233  }
234 
235  real_time = duration_cast<milliseconds>(high_resolution_clock::now() - process_start_time_point);
236 
237  constexpr auto timevalToChrono = [](const timeval & val) -> std::chrono::milliseconds
238  {
239  using namespace std::chrono;
240  return duration_cast<milliseconds>(seconds(val.tv_sec) + microseconds(val.tv_usec));
241  };
242 
243  cpu_time = timevalToChrono(resource_usage.ru_utime) + timevalToChrono(resource_usage.ru_stime);
244  memory = kerbal::utility::KB(resource_usage.ru_maxrss);
245  exit_code = WEXITSTATUS(status);
246  }
247 
248 
249  if (exit_code != 0) {
250  running_result = ProtectedProcessResult::RUNTIME_ERROR;
251  return ProtectedProcessDetails(running_result, real_time, cpu_time, memory, exit_code);
252  }
253 
254  // if signaled
255  if (WIFSIGNALED(status) != 0) {
256  LOG_DEBUG(jobType, s_id, log_fp, "signal: ", WTERMSIG(status));
257  switch (WTERMSIG(status)) {
258  case SIGSEGV:
259  if (config.max_memory.has_value() && memory > config.max_memory.value()) {
260  running_result = ProtectedProcessResult::MEMORY_LIMIT_EXCEEDED;
261  } else {
262  running_result = ProtectedProcessResult::RUNTIME_ERROR;
263  }
264  break;
265  case SIGUSR1:
266  running_result = ProtectedProcessResult::SYSTEM_ERROR;
267  break;
268  default:
269  running_result = ProtectedProcessResult::RUNTIME_ERROR;
270  }
271  } else {
272  if (config.max_memory.has_value() && memory > config.max_memory.value()) {
273  running_result = ProtectedProcessResult::MEMORY_LIMIT_EXCEEDED;
274  }
275  }
276  if (config.max_real_time.has_value() && real_time > config.max_real_time.value()) {
277  running_result = ProtectedProcessResult::REAL_TIME_LIMIT_EXCEEDED;
278  }
279  if (config.max_cpu_time.has_value() && cpu_time > config.max_cpu_time.value()) {
280  running_result = ProtectedProcessResult::CPU_TIME_LIMIT_EXCEEDED;
281  }
282  return ProtectedProcessDetails(running_result, real_time, cpu_time, memory, exit_code);
283 }
