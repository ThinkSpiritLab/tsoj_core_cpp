87 void SingleTestJudgeJob::store_source_code_procedure()
88 {
89  kerbal::redis_v2::reply reply;
90 
91  try {
92  kerbal::redis_v2::connection redis_conn;
93  reply = redis_conn.execute("hget source_code:%d:%d source", this->jobType, this->s_id.to_literal());
94  } catch (const std::exception & e) {
95  throw std::runtime_error("Get source code failed!");
96  }
97 
98  if (reply.type() != kerbal::redis_v2::reply_type::STRING) {
99  throw std::runtime_error("Get source code failed!");
100  }
101 
102  try {
103  this->languageStrategy->save_source_code(this->working_dir, "Main", reply->str);
104  } catch (const std::exception & e) {
105  throw std::runtime_error("Save source code failed!");
106  }

109 void SingleTestJudgeJob::compile_procedure(UnitedJudgeResult & judge_result)
110 {
111  if (judge_result != UnitedJudgeResult::ACCEPTED) {
112  return;
113  }
114 
115  using namespace kerbal::compatibility::chrono_suffix;
116  using namespace kerbal::utility;
117  using namespace std::string_literals;
118 
119  ProtectedProcessConfig compile_config("/dev/null", this->working_dir / "compiler.out", this->working_dir / "compiler.out");
120  compile_config.set_max_cpu_time(15_s);
121  compile_config.set_max_real_time(15_s);
122  compile_config.set_max_stack(1024_MB);
123  compile_config.set_max_output_size(128_MB);
124 
125  boost::filesystem::path source_file_path = this->working_dir / ("Main."s + languageStrategy->default_suffix());
126  const auto [running_result, real_time, cpu_time, memory, exit_code] =
127  languageStrategy->compile(source_file_path, compile_config);
128 
129  switch (running_result) {
130  case ProtectedProcessResult::ACCEPTED:
131  break;
132  case ProtectedProcessResult::REAL_TIME_LIMIT_EXCEEDED:
133  case ProtectedProcessResult::CPU_TIME_LIMIT_EXCEEDED:
134  case ProtectedProcessResult::MEMORY_LIMIT_EXCEEDED:
135  case ProtectedProcessResult::OUTPUT_LIMIT_EXCEEDED:
136  judge_result = UnitedJudgeResult::COMPILE_RESOURCE_LIMIT_EXCEED;
137  break;
138  case ProtectedProcessResult::RUNTIME_ERROR:
139  judge_result = UnitedJudgeResult::COMPILE_ERROR;
140  break;
141  case ProtectedProcessResult::SYSTEM_ERROR:
142  judge_result = UnitedJudgeResult::SYSTEM_ERROR;
143  break;
144  }
145 }

147 void SingleTestJudgeJob::running_procedure(UnitedJudgeResult & judge_result, ProtectedProcessDetails & running_details)
148 {
149  if (judge_result != UnitedJudgeResult::ACCEPTED) {
150  return;
151  }
152 
153  using namespace kerbal::compatibility::chrono_suffix;
154  using namespace kerbal::utility;
155  using namespace std::string_literals;
156 
157  boost::filesystem::path input_file = Settings::JudgeSettings::input_dir() / to_string(p_id) / "in.1";
158  ProtectedProcessConfig running_config(input_file, this->working_dir / "user.out", "/dev/null");
159  running_config.set_max_cpu_time(this->time_limit);
160  running_config.set_max_real_time(this->time_limit.value_or(15_s) + 1_s);
161  running_config.set_max_stack(this->memory_limit);
162  running_config.set_max_memory(this->memory_limit);
163  running_config.set_max_output_size(32_MB);
164 
165  running_details = languageStrategy->running(this->working_dir / "Main", running_config);
166 
167  switch (running_details.running_result()) {
168  case ProtectedProcessResult::ACCEPTED:
169  judge_result = UnitedJudgeResult::WRONG_ANSWER;
170 // judge_result = compare();
171  break;
172  case ProtectedProcessResult::REAL_TIME_LIMIT_EXCEEDED:
173  judge_result = UnitedJudgeResult::REAL_TIME_LIMIT_EXCEEDED;
174  break;
175  case ProtectedProcessResult::CPU_TIME_LIMIT_EXCEEDED:

177  break;
178  case ProtectedProcessResult::MEMORY_LIMIT_EXCEEDED:
179  judge_result = UnitedJudgeResult::MEMORY_LIMIT_EXCEEDED;
180  break;

182  judge_result = UnitedJudgeResult::OUTPUT_LIMIT_EXCEEDED;
183  break;

185  judge_result = UnitedJudgeResult::RUNTIME_ERROR;
