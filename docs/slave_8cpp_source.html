193 void listenning_loop()
194 {
195  auto redis_conn_handler = sync_fetch_redis_conn();
196  kerbal::redis_v2::connection & main_conn = *redis_conn_handler;
197  kerbal::redis_v2::list judge_queue(main_conn, "judge_queue");
198 
199  while (loop) {
200 
201  // Step 1: Pop job item and parser job_type and job_id.
202  int job_type(0);
203  ojv4::s_id_type job_id(0);
204 
205  /*
206  * 当收到 SIGTERM 信号时，会在评测队列末端加一个特殊的评测任务用于标识停止测评。此处若检测到停止
207  * 工作的信号，则结束工作loop
208  * 若取得的是正常的评测任务则继续工作
209  */
210  try {
211  std::string job_item = judge_queue.blpop(0_s);
212  try {
213  std::tie(job_type, job_id) = JudgeJob::parseJobItem(job_item);
214  } catch (const std::exception & e) {
215  EXCEPT_FATAL(0, 0, log_fp, "Fail to parse job item.", e, "job_item: ", job_item);
216  continue;
217  }
218  LOG_INFO(job_type, job_id, log_fp, "<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<");
219  LOG_INFO(job_type, job_id, log_fp, "Judge server: ", judge_server_id, " get judge job: ", job_item);
220  } catch (const std::exception & e) {
221  EXCEPT_FATAL(0, 0, log_fp, "Fail to fetch job.", e);
222  continue;
223  } catch (...) {
224  UNKNOWN_EXCEPT_FATAL(0, 0, log_fp, "Fail to fetch job.");
225  continue;
226  }
227 
228  /*
229  * 当子进程数达到最大子进程数设定时，等待一个子进程结束后再运行。
230  *
231  * 补充一个注释作者一开始没有想明白的地方，觉得有必要在此说明。当 loop 循环人为结束之后，有可能存在部分
232  * 评测子进程并没有结束。因此失去了在此使用 waitpid 获取状态信息并释放的机会。但是这并不会造成僵死进程
233  * 的堆积。当父进程结束之后，子进程的父进程会变为 init 进程，即孤儿进程被收养，而孤儿进程结束之后，会自动调用
234  * waitpid 并释放，因此并不会存在僵死进程堆积。
235  */
236  pid_t pid;
237  while ((pid = waitpid(-1, NULL, WNOHANG)) > 0) {
238  cur_running--;
239  }
240  if (cur_running >= max_running) {
241  pid = waitpid(-1, NULL, 0);
242  cur_running--;
243  }
244 
245  try {
246  process judge_process([](int job_type, ojv4::s_id_type job_id) {
247  RedisContext child_conn(redis_hostname.c_str(), redis_port, 1500_ms);
248  if (!child_conn) {
249  LOG_FATAL(job_type, job_id, log_fp, "Child context connect failed.");
250  //TODO 插入到 failure list 中
251  return;
252  }
253 
254  std::unique_ptr<JudgeJob> job = nullptr;
255 
256  try {
257  job.reset(new JudgeJob(job_type, job_id, child_conn));
258  } catch (const std::exception & e) {
259  EXCEPT_FATAL(job_type, job_id, log_fp, "Fail to construct job.", e);
260  return;
261  } catch (...) {
262  UNKNOWN_EXCEPT_FATAL(job_type, job_id, log_fp, "Fail to construct job");
263  return;
264  }
265 
266  // Step 2: Judge job.
267  try {
268  job->handle();
269  } catch (const std::exception & e) {
270  EXCEPT_FATAL(job_type, job_id, log_fp, "Fail to judge job.", e);
271  job->push_back_failed_judge_job();
272  return;
273  } catch (...) {
274  UNKNOWN_EXCEPT_FATAL(job_type, job_id, log_fp, "Fail to judge job.");
275  job->push_back_failed_judge_job();
276  return;
277  }
278 
279  LOG_INFO(job_type, job_id, log_fp, "Judge finished.");
280  LOG_INFO(job_type, job_id, log_fp, ">>>>>>>>>>>>>>>>>>>>>>>>>>>>>>");
281 
282  }, job_type, job_id);
283  judge_process.detach();
284  LOG_DEBUG(job_type, job_id, log_fp, "Judge process fork success. Child_pid: ", judge_process.get_child_id());
285  ++cur_running;
286  } catch (const std::exception & e) {
287  EXCEPT_FATAL(job_type, job_id, log_fp, "Judge process fork failed.", e);
288  JudgeJob::insert_into_failed(main_conn, job_type, job_id);
289  continue;
290  } catch (...) {
291  UNKNOWN_EXCEPT_FATAL(job_type, job_id, log_fp, "Judge process fork failed.");
292  JudgeJob::insert_into_failed(main_conn, job_type, job_id);
293  continue;
294  }
295  } // loop
296 }
