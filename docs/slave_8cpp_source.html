112 void register_self() noexcept try
113 {
114  using namespace std::chrono;
115  using namespace kerbal::compatibility::chrono_suffix;
116 
117  static auto regist_self_conn = context_pool->apply(getpid());
118  LOG_INFO(0, 0, log_fp, "Regist_self service get context: ", regist_self_conn.get_id());
119 
120  static Operation opt(regist_self_conn);
121  while (true) {
122  try {
123  const time_t now = time(NULL);
124  const std::string confirm_time = get_ymd_hms_in_local_time_zone(now);
125  opt.hmset("judge_server:" + judge_server_id,
126  "host_name"_cptr, host_name,
127  "ip"_cptr, ip,
128  "user_name"_cptr, user_name,
129  "listening_pid"_cptr, listening_pid,
130  "last_confirm"_cptr, confirm_time);
131  opt.set("judge_server_confirm:" + judge_server_id, confirm_time);
132  opt.expire("judge_server_confirm:" + judge_server_id, 30_s);
133 
134  static Set<std::string> s(regist_self_conn, "online_judger");
135  s.insert(judge_server_id);
136 
137  LOG_DEBUG(0, 0, log_fp, "Register self success");
138  std::this_thread::sleep_for(10_s);
139  } catch (const RedisException & e) {
140  EXCEPT_FATAL(0, 0, log_fp, "Register self failed.", e);
141  }
142  }
143 } catch (...) {
144  UNKNOWN_EXCEPT_FATAL(0, 0, log_fp, "Register self failed.");
145  exit(2);

249 int main(int argc, const char * argv[]) try
250 {
251  using namespace kerbal::compatibility::chrono_suffix;
252 
253  if (argc > 1 && argv[1] == std::string("--version")) {
254  cout << "version: " __DATE__ " " __TIME__ << endl;
255  exit(0);
256  }
257  cout << std::boolalpha;
258  load_config();
259 
260  try {
261  // +2 : 除了四个评测进程需要 redis 连接以外, 监听队列, 发送心跳进程也各需要一个
262  // 当前版本中连接池并未启用
263  context_pool.reset(new RedisContextPool(max_running + 2, redis_hostname.c_str(), redis_port, 1500_ms));
264  } catch (const std::exception & e) {
265  EXCEPT_FATAL(0, 0, log_fp, "Redis connect failed!", e);
266  exit(-2);
267  }
268 
269  auto main_conn = context_pool->apply(getpid());
270  LOG_INFO(0, 0, log_fp, "main_conn: ", main_conn.get_id());
271 
272  try {
276  std::thread regist(register_self);
277  regist.detach();
278  } catch (const std::exception & e) {
279  LOG_FATAL(0, 0, log_fp, "Register self service process fork failed.");
280  exit(-3);
281  }
282 
283  signal(SIGTERM, regist_SIGTERM_handler);
284 
285  List<std::string> job_list(main_conn, "judge_queue");
286 
287  while (loop) {
288 
289  // Step 1: Pop job item and parser job_type and job_id.
290  std::string job_item = "0,0";
291  int job_type = 0;
292  int job_id = 0;
293  /*
294  * 当收到 SIGTERM 信号时，会在评测队列末端加一个特殊的评测任务用于标识停止测评。此处若检测到停止
295  * 工作的信号，则结束工作loop
296  * 若取得的是正常的评测任务则继续工作
297  */
298  try {
299  job_item = job_list.block_pop_front(0_s);
300  if (JobBase::isExitJob(job_item) == true) {
301  loop = false;
302  LOG_INFO(0, 0, log_fp, "Get exit job.");
303  continue;
304  }
305  std::tie(job_type, job_id) = JudgeJob::parseJobItem(job_item);
306  LOG_DEBUG(job_type, job_id, log_fp, "Judge server ", judge_server_id, " get job: ", job_item);
307  } catch (const RedisNilException & e) {
308  EXCEPT_FATAL(0, 0, log_fp, "Fail to fetch job.", e, ", job_item: ", job_item);
309  continue;
310  } catch (const std::exception & e) {
311  EXCEPT_FATAL(0, 0, log_fp, "Fail to fetch job.", e, ", job_item: ", job_item);
312  continue;
313  } catch (...) {
314  UNKNOWN_EXCEPT_FATAL(0, 0, log_fp, "Fail to fetch job.", ", job_item: ", job_item);
315  continue;
316  }
317 
318  /*
319  * 当子进程数达到最大子进程数设定时，等待一个子进程结束后再运行。
320  *
321  * 补充一个注释作者一开始没有想明白的地方，觉得有必要在此说明。当 loop 循环人为结束之后，有可能存在部分
322  * 评测子进程并没有结束。因此失去了在此使用 waitpid 获取状态信息并释放的机会。但是这并不会造成僵死进程
323  * 的堆积。当父进程结束之后，子进程的父进程会变为 init 进程，即孤儿进程被收养，而孤儿进程结束之后，会自动调用
324  * waitpid 并释放，因此并不会存在僵死进程堆积。
325  */
326  pid_t pid;
327  while ((pid = waitpid(-1, NULL, WNOHANG)) > 0) {
328  cur_running--;
329  }
330  if (cur_running >= max_running) {
331  pid = waitpid(-1, NULL, 0);
332  cur_running--;
333  }
334 
335  try {
336  Process judge_process(false, [](int job_type, int job_id) {
337  RedisContext child_conn(redis_hostname.c_str(), redis_port, 1500_ms);
338  if (!child_conn) {
339  LOG_FATAL(job_type, job_id, log_fp, "Child context connect failed.");
340  //TODO 插入到 failure list 中
341  return;
342  }
343 
344  std::unique_ptr<JudgeJob> job = nullptr;
345 
346  try {
347  job.reset(new JudgeJob(job_type, job_id, child_conn));
348  } catch (const std::exception & e) {
349  EXCEPT_FATAL(job_type, job_id, log_fp, "Fail to create job.", e);
350  return;
351  } catch (...) {
352  UNKNOWN_EXCEPT_FATAL(job_type, job_id, log_fp, "Fail to create job");
353  return;
354  }
355 
356  // Step 2: Judge job.
357  try {
358  job->handle();
359  } catch (const std::exception & e) {
360  EXCEPT_FATAL(job_type, job_id, log_fp, "Fail to judge job.", e);
361  job->push_back_failed_judge_job();
362  return;
363  } catch (...) {
364  UNKNOWN_EXCEPT_FATAL(job_type, job_id, log_fp, "Fail to judge job.");
365  job->push_back_failed_judge_job();
366  return;
367  }
368  }, job_type, job_id);
369  LOG_DEBUG(job_type, job_id, log_fp, "Judge process fork success. Child_pid: ", judge_process.get_child_id());
370  ++cur_running;
371  } catch (const std::exception & e) {
372  EXCEPT_FATAL(job_type, job_id, log_fp, "Judge process fork failed.", e);
373  JudgeJob::insert_into_failed(main_conn, job_type, job_id);
374  continue;
375  } catch (...) {
376  UNKNOWN_EXCEPT_FATAL(job_type, job_id, log_fp, "Judge process fork failed.");
377  JudgeJob::insert_into_failed(main_conn, job_type, job_id);
378  continue;
379  }
380  } // loop
381  LOG_INFO(0, 0, log_fp, "Judge server exit."_cptr);
382  return 0;
383 
384 } catch (const std::exception & e) {
385  EXCEPT_FATAL(0, 0, log_fp, "An uncaught exception caught by main.", e);
386  throw;
387 } catch (...) {
388  UNKNOWN_EXCEPT_FATAL(0, 0, log_fp, "An uncaught exception caught by main.");
389  throw;
390 }
