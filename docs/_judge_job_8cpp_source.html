44  this->storeSourceCode();

51  Result compile_result = this->compile();
52  LOG_DEBUG(jobType, sid, log_fp, "compile finished. compile result: ", compile_result);
53 
54  switch (compile_result.judge_result) {
55  case UnitedJudgeResult::ACCEPTED: {
56  LOG_DEBUG(jobType, sid, log_fp, "compile success");

58  Result result = this->running();
59  LOG_DEBUG(jobType, sid, log_fp, "judge finished");
60  this->commitJudgeResultToRedis(result);
61  break;
62  }
63  case UnitedJudgeResult::COMPILE_RESOURCE_LIMIT_EXCEED:
64  LOG_WARNING(jobType, sid, log_fp, "compile resource limit exceed");
65  this->commitJudgeResultToRedis(compile_result);
66  break;
67  case UnitedJudgeResult::SYSTEM_ERROR:
68  LOG_DEBUG(jobType, sid, log_fp, "system error");
69  this->commitJudgeResultToRedis(compile_result);
70  break;
71  case UnitedJudgeResult::COMPILE_ERROR:
72  default:
73  LOG_DEBUG(jobType, sid, log_fp, "compile failed");
74  this->commitJudgeResultToRedis(compile_result);
75 
76  if (this->set_compile_info() == false) {
77  LOG_WARNING(jobType, sid, log_fp, "An error occurred while getting compile info.");
78  }
79  break;
80  }
81 
82  //update update_queue
83  static RedisCommand update_queue("rpush update_queue %d,%d");
84  update_queue.execute(redisConn, jobType, sid);
85 
86 #ifndef DEBUG
87  // clear the dir
88  this->clean_job_dir();
89 #endif

165 bool JudgeJob::set_compile_info() noexcept
166 {
167  std::ifstream fin("compiler.out", std::ios::in);
168  if (!fin) {
169  LOG_FATAL(jobType, sid, log_fp, "Cannot open compiler.out");
170  return false;
171  }
172 
173  // get length of file:
174  fin.seekg(0, fin.end);
175  int length = fin.tellg();
176  fin.seekg(0, fin.beg);
177 
178  constexpr int MYSQL_TEXT_MAX_SIZE = 65535;
179  char buffer[MYSQL_TEXT_MAX_SIZE + 10];
180  fin.read(buffer, MYSQL_TEXT_MAX_SIZE);
181 
182  if (fin.bad()) {
183  LOG_FATAL(jobType, sid, log_fp, "Read compile info error, only ", fin.gcount(), " could be read");
184  LOG_FATAL(jobType, sid, log_fp, "The read buffer: ", buffer);
185  return false;
186  }
187 
188  // Mysql 中存储字符长度有限，此处似乎超出长度可能导致的乱码问题未解决
189  if (MYSQL_TEXT_MAX_SIZE >= 3 && length > MYSQL_TEXT_MAX_SIZE) {
190  char * end = buffer + MYSQL_TEXT_MAX_SIZE - 3;
191  for (int i = 0; i < 3; ++i) {
192  *end = '.';
193  ++end;
194  }
195  }
196 
197  try {
198  static boost::format compile_info("compile_info:%d:%d");
199  Operation opt(redisConn);
200  opt.set((compile_info % jobType % sid).str(), buffer);
201  } catch (const std::exception & e) {
202  EXCEPT_FATAL(jobType, sid, log_fp, "Set compile info failed.", e);
203  return false;
204  }
205 
206  return true;
207 }

255 Result JudgeJob::execute(const Config & config) const noexcept
256 {
257  using namespace std::chrono;
258  using namespace kerbal::compatibility::chrono_suffix;
259 
260  Result result;
261 
262  // check whether current user is root
263  if (getuid() != 0) {
264  LOG_FATAL(jobType, sid, log_fp, "ROOT_REQUIRED");
265  result.setErrorCode(RunnerError::ROOT_REQUIRED);
266  return result;
267  }
268 
269  // check args
270  if (config.check_is_valid_config() == false) {
271  LOG_FATAL(jobType, sid, log_fp, "INVALID_CONFIG");
272  result.setErrorCode(RunnerError::INVALID_CONFIG);
273  return result;
274  }
275 
276  std::unique_ptr<Process> child_process = nullptr;
277 
278  // 此处分离了一个运行子进程，稍后该进程内自我完成工作配置，然后替换为用户编写的程序，用于实际的编译/运行
279  try {
280  child_process.reset(new Process(true, [this, &config]() {
281  this->child_process(config);
282  }));
283  } catch (const std::exception & e) {
284  LOG_FATAL(jobType, sid, log_fp, "Fork failed. Error information: ", e.what());
285  result.setErrorCode(RunnerError::FORK_FAILED);
286  return result;
287  }
288 
289  // record current time
290  time_point<high_resolution_clock> start(high_resolution_clock::now());
291 
292  if (config.limitRealTime()) {
293  std::unique_ptr<std::thread> timeout_killer_thread = nullptr;
294  try {
295  timeout_killer_thread.reset(new std::thread([&child_process](milliseconds timeout) {
296  std::this_thread::sleep_for(timeout + 1_s);
297  if (child_process->kill(SIGKILL) != 0) {
298  //TODO log
299  }
300  }, config.max_real_time));
301  } catch (const std::system_error & e) {
302  child_process->kill(SIGKILL);
303  EXCEPT_FATAL(jobType, sid, log_fp, "Thread construct failed.", e, " , error code: ", e.code().value());
304  result.setErrorCode(RunnerError::PTHREAD_FAILED);
305  return result;
306  } catch (...) {
307  child_process->kill(SIGKILL);
308  UNKNOWN_EXCEPT_FATAL(jobType, sid, log_fp, "Thread construct failed.");
309  result.setErrorCode(RunnerError::PTHREAD_FAILED);
310  return result;
311  }
312 
313  try {
314  timeout_killer_thread->detach();
315  } catch (const std::system_error & e) {
316  child_process->kill(SIGKILL);
317  EXCEPT_FATAL(jobType, sid, log_fp, "Thread detach failed.", e, " , error code: ", e.code().value());
318  result.setErrorCode(RunnerError::PTHREAD_FAILED);
319  return result;
320  } catch (...) {
321  child_process->kill(SIGKILL);
322  UNKNOWN_EXCEPT_FATAL(jobType, sid, log_fp, "Thread detach failed.");
323  result.setErrorCode(RunnerError::PTHREAD_FAILED);
324  return result;
325  }
326  LOG_DEBUG(jobType, sid, log_fp, "Timeout thread work success");
327  }
328 
329  int status;
330  struct rusage resource_usage;
331 
332  // wait for child process to terminate
333  // on success, returns the process ID of the child whose state has changed;
334  // On error, -1 is returned.
335  if (child_process->wait4(&status, WSTOPPED, &resource_usage) == -1) {
336  child_process->kill(SIGKILL);
337  LOG_FATAL(jobType, sid, log_fp, "WAIT_FAILED");
338  result.setErrorCode(RunnerError::WAIT_FAILED);
339  return result;
340  }
341 
342  result.real_time = duration_cast<milliseconds>(high_resolution_clock::now() - start);
343  result.cpu_time = timevalToChrono(resource_usage.ru_utime) + timevalToChrono(resource_usage.ru_stime);
344  result.memory = kerbal::utility::KB(resource_usage.ru_maxrss);
345  result.exit_code = WEXITSTATUS(status);
346 
347  if (result.exit_code != 0) {
348  result.judge_result = UnitedJudgeResult::RUNTIME_ERROR;
349  return result;
350  }
351  // if signaled
352  if (WIFSIGNALED(status) != 0) {
353  result.signal = WTERMSIG(status);
354  LOG_DEBUG(jobType, sid, log_fp, "signal: ", result.signal);
355  switch (result.signal) {
356  case SIGSEGV:
357  if (config.limitMemory() && result.memory > config.max_memory) {
358  result.judge_result = UnitedJudgeResult::MEMORY_LIMIT_EXCEEDED;
359  } else {
360  result.judge_result = UnitedJudgeResult::RUNTIME_ERROR;
361  }
362  break;
363  case SIGUSR1:
364  result.judge_result = UnitedJudgeResult::SYSTEM_ERROR;
365  break;
366  default:
367  result.judge_result = UnitedJudgeResult::RUNTIME_ERROR;
368  }
369  } else {
370  if (config.limitMemory() && result.memory > config.max_memory) {
371  result.judge_result = UnitedJudgeResult::MEMORY_LIMIT_EXCEEDED;
372  }
373  }
374  if (config.limitRealTime() && result.real_time > config.max_real_time) {
375  result.judge_result = UnitedJudgeResult::REAL_TIME_LIMIT_EXCEEDED;
376  }
377  if (config.limitCPUTime() && result.cpu_time > config.max_cpu_time) {
378  result.judge_result = UnitedJudgeResult::CPU_TIME_LIMIT_EXCEEDED;
379  }
380  return result;
381 }

383 Result JudgeJob::compile() const noexcept
384 {
385  /*
386  * 与 running 中不同，运行测试样例可能需要多次，而编译只需执行一次，因此只需要生成一个匿名的 CompileConfig ，节约了传递成本
387  * （也好看
388  */
389  Result result = this->execute(CompileConfig(*this));
390 
391  switch (result.judge_result) {
392  case UnitedJudgeResult::ACCEPTED:
393  return result;
394  case UnitedJudgeResult::CPU_TIME_LIMIT_EXCEEDED:
395  case UnitedJudgeResult::REAL_TIME_LIMIT_EXCEEDED:
396  case UnitedJudgeResult::MEMORY_LIMIT_EXCEEDED:
397  result.judge_result = UnitedJudgeResult::COMPILE_RESOURCE_LIMIT_EXCEED;
398  return result;
399  case UnitedJudgeResult::RUNTIME_ERROR:
400  result.judge_result = UnitedJudgeResult::COMPILE_ERROR;
401  return result;
402  case UnitedJudgeResult::SYSTEM_ERROR:
403  LOG_FATAL(jobType, sid, log_fp, "System error occurred while compiling. Execute result: ", result);
404  return result;
405  default:
406  LOG_FATAL(jobType, sid, log_fp, "Unexpected compile result: ", result);
407  result.judge_result = UnitedJudgeResult::COMPILE_ERROR;
408  return result;
409  }

412 bool JudgeJob::child_process(const Config & config) const
413 {
414  struct rlimit max_stack;
415  max_stack.rlim_cur = max_stack.rlim_max = (rlim_t) (config.max_stack.count());
416  if (setrlimit(RLIMIT_STACK, &max_stack) != 0) {
417  LOG_FATAL(jobType, sid, log_fp, RunnerError::SETRLIMIT_FAILED);
418  raise(SIGUSR1);
419  return false;
420  }
421 
422  // set memory limit
423  if (config.limitMemory()) {
424  struct rlimit max_memory;
425  max_memory.rlim_cur = max_memory.rlim_max = (rlim_t) (config.max_memory.count()) * 2;
426  if (setrlimit(RLIMIT_AS, &max_memory) != 0) {
427  LOG_FATAL(jobType, sid, log_fp, RunnerError::SETRLIMIT_FAILED);
428  raise(SIGUSR1);
429  return false;
430  }
431  }
432 
433  // set cpu time limit (in seconds)
434  if (config.limitCPUTime()) {
435  struct rlimit max_cpu_time;
436  max_cpu_time.rlim_cur = max_cpu_time.rlim_max = (rlim_t) ((config.max_cpu_time.count() + 1000) / 1000);
437  if (setrlimit(RLIMIT_CPU, &max_cpu_time) != 0) {
438  LOG_FATAL(jobType, sid, log_fp, RunnerError::SETRLIMIT_FAILED);
439  raise(SIGUSR1);
440  return false;
441  }
442  }
443 
444  // set max process number limit
445  if (config.limitProcessNumber()) {
446  struct rlimit max_process_number;
447  max_process_number.rlim_cur = max_process_number.rlim_max = (rlim_t) config.max_process_number;
448  if (setrlimit(RLIMIT_NPROC, &max_process_number) != 0) {
449  LOG_FATAL(jobType, sid, log_fp, RunnerError::SETRLIMIT_FAILED);
450  raise(SIGUSR1);
451  return false;
452  }
453  }
454 
455  // set max output size limit
456  if (config.limitOutput()) {
457  struct rlimit max_output_size;
458  max_output_size.rlim_cur = max_output_size.rlim_max = (rlim_t) config.max_output_size.count();
459  if (setrlimit(RLIMIT_FSIZE, &max_output_size) != 0) {
460  LOG_FATAL(jobType, sid, log_fp, RunnerError::SETRLIMIT_FAILED);
461  raise(SIGUSR1);
462  return false;
463  }
464  }
465 
466  std::unique_ptr<FILE, int (*)(FILE *)> input_file(fopen(config.input_path, "r"), fclose);
467  if (config.input_path != nullptr) {
468  if (!input_file) {
469  LOG_FATAL(jobType, sid, log_fp, "can not open [", config.input_path, "]");
470  raise(SIGUSR1);
471  return false;
472  }
473  // redirect file -> stdin
474  // On success, these system calls return the new descriptor.
475  // On error, -1 is returned, and errno is set appropriately.
476  if (dup2(fileno(input_file.get()), fileno(stdin)) == -1) {
477  LOG_FATAL(jobType, sid, log_fp, RunnerError::DUP2_FAILED, "dup2 input file failed");
478  raise(SIGUSR1);
479  return false;
480  }
481  }
482 
483  /*
484  * Out | Err
485  * -------------------------
486  * null | null
487  * null | err.txt
488  * out.txt | null
489  * out.txt | out.txt (DO NOT open twice)
490  * out.txt | err.txt
491  */
492  std::shared_ptr<FILE> output_file = nullptr, error_file = nullptr;
493 
494  if (config.output_path != nullptr && config.error_path != nullptr &&
495  strcmp(config.output_path, config.error_path) == 0) {
496  // 标准输出与错误输出指定为同一个文件
497  auto of_ptr = fopen(config.output_path, "w"); //DO NOT fclose this ptr while exit brace
498  if (!of_ptr) {
499  LOG_FATAL(jobType, sid, log_fp, "can not open \"", config.output_path, "\"");
500  raise(SIGUSR1);
501  return false;
502  } else {
503  output_file.reset(of_ptr, fclose);
504  error_file = output_file;
505  }
506 
507  } else {
508  if (config.output_path != nullptr) {
509  auto of_ptr = fopen(config.output_path, "w"); //DO NOT fclose this ptr while exit brace
510  if (!of_ptr) {
511  LOG_FATAL(jobType, sid, log_fp, "can not open [", config.output_path, "]");
512  raise(SIGUSR1);
513  return false;
514  } else {
515  output_file.reset(of_ptr, fclose);
516  }
517  }
518 
519  if (config.error_path != nullptr) {
520  auto ef_ptr = fopen(config.error_path, "w"); //DO NOT fclose this ptr while exit brace
521  if (!ef_ptr) {
522  LOG_FATAL(jobType, sid, log_fp, "can not open [", config.error_path, "]");
523  raise(SIGUSR1);
524  return false;
525  } else {
526  error_file.reset(ef_ptr, fclose);
527  }
528  }
529  }
530 
531  // redirect stdout -> file
532  if (output_file != nullptr) {
533  if (dup2(fileno(output_file.get()), fileno(stdout)) == -1) {
534  LOG_FATAL(jobType, sid, log_fp, RunnerError::DUP2_FAILED, "dup2 output file failed");
535  raise(SIGUSR1);
536  return false;
537  }
538  }
539  if (error_file != nullptr) {
540  // redirect stderr -> file
541  if (dup2(fileno(error_file.get()), fileno(stderr)) == -1) {
542  LOG_FATAL(jobType, sid, log_fp, RunnerError::DUP2_FAILED, "dup2 error file failed");
543  raise(SIGUSR1);
544  return false;
545  }
546  }
547 
548  // set gid
549  gid_t group_list[] = {config.gid};
550  if (setgid(config.gid) == -1 || setgroups(sizeof(group_list) / sizeof(gid_t), group_list) == -1) {
551  LOG_FATAL(jobType, sid, log_fp, RunnerError::SETUID_FAILED);
552  raise(SIGUSR1);
553  return false;
554  }
555 
556  // set uid
557  if (setuid(config.uid) == -1) {
558  LOG_FATAL(jobType, sid, log_fp, RunnerError::SETUID_FAILED);
559  raise(SIGUSR1);
560  return false;
561  }
562 
563  // load seccomp
564  if (config.load_seccomp_rules() != RunnerError::SUCCESS) {
565  LOG_FATAL(jobType, sid, log_fp, RunnerError::LOAD_SECCOMP_FAILED);
566  raise(SIGUSR1);
567  return false;
568  }
569 
570  auto args = config.args.getArgs();
571  auto env = config.env.getArgs();
572  execve(config.exe_path, args.get(), env.get());
573  /*
574  * 以上三行不可以并作
575  * execve(config.exe_path, config.args.getArgs().get(), config.env.getArgs().get());
576  * 会取到野指针
577  */
578 
579  LOG_FATAL(jobType, sid, log_fp, RunnerError::EXECVE_FAILED);
580  raise(SIGUSR1);
581  return false;
582 }
