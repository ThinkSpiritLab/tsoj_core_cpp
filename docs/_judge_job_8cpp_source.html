54 void JudgeJob::handle()
55 {
56  LOG_DEBUG(jobType, s_id, log_fp, "JudgeJob::handle");
57 
58  this->change_job_dir();
59  this->storeSourceCode();
60 
61  LOG_DEBUG(jobType, s_id, log_fp, "store source code finished");
62 
63  int similarity_percentage = -1;
64  try {
65  similarity_percentage = this->calculate_similarity();
66  } catch (const std::exception & e) {
67  EXCEPT_FATAL(jobType, s_id, log_fp, "An error occurred while calculate similarity.", e);
68  //DO NOT THROW
69  }
70 
71  LOG_DEBUG(jobType, s_id, log_fp, "similarity: ", similarity_percentage);
72 
73  if (this->commit_similarity_details_to_redis() == false) {
74  LOG_FATAL(jobType, s_id, log_fp, "An error occurred while commit similarity details to redis.");
75  }
76 
77  SolutionDetails judgeResult;
78  if (this->similarity_threshold != 0 && !this->have_accepted &&
79  similarity_percentage > this->similarity_threshold) {
80  judgeResult.judge_result = UnitedJudgeResult::SIMILAR_CODE;
81  } else {
82 
83  // compile
84  this->commitJudgeStatusToRedis(JudgeStatus::COMPILING);
85  LOG_DEBUG(jobType, s_id, log_fp, "compile start");
86  judgeResult = static_cast<SolutionDetails>(this->compile());
87  LOG_DEBUG(jobType, s_id, log_fp, "compile finished. compile result: ", judgeResult);
88 
89  switch (judgeResult.judge_result) {
90  case UnitedJudgeResult::ACCEPTED: {
91  LOG_DEBUG(jobType, s_id, log_fp, "compile success");
92  // 编译成功则 run
93  judgeResult = static_cast<SolutionDetails>(this->running());
94  LOG_DEBUG(jobType, s_id, log_fp, "judge finished");
95 
96  // 如果用户 AC 且要求留存代码, 则将代码保存至留存目录
97  if (judgeResult.judge_result == UnitedJudgeResult::ACCEPTED && this->no_store_ac_code != true) {
98  this->storeSourceCode("%s/%d/%d/"_fmt(accepted_solutions_dir, p_id, lang == Language::Java ? 2 : 0).str(), std::to_string(s_id.to_literal()));
99  }
100  break;
101  }
102  case UnitedJudgeResult::COMPILE_RESOURCE_LIMIT_EXCEED:
103  LOG_WARNING(jobType, s_id, log_fp, "compile resource limit exceed");
104  break;
105  case UnitedJudgeResult::SYSTEM_ERROR:
106  LOG_FATAL(jobType, s_id, log_fp, "system error");
107  break;
108  case UnitedJudgeResult::COMPILE_ERROR:
109  default:
110  LOG_DEBUG(jobType, s_id, log_fp, "compile failed");
111 
112  if (this->set_compile_info() == false) {
113  LOG_WARNING(jobType, s_id, log_fp, "An error occurred while getting compile info.");
114  }
115  break;
116  }
117  }
118 
119  LOG_DEBUG(jobType, s_id, log_fp, "judge result: ", judgeResult);
120  this->commitJudgeResultToRedis(judgeResult, similarity_percentage);
121 
122  //update update_queue
123  RedisCommand update_queue("rpush update_queue %d,%d");
124  update_queue.execute(redisConn, jobType, s_id);
125 
126 #ifndef DEBUG
127  // clear the dir
128  this->clean_job_dir();
129 #endif

133 int JudgeJob::calculate_similarity()
134 {
135  std::string sim_cmd;
136  switch (this->lang) {
137  case Language::C:
138  sim_cmd = (boost::format("sim_c -S -P Main.c / %s/%d/0/* > sim.txt") % accepted_solutions_dir % this->p_id).str();
139  break;
140  case Language::Cpp:
141  case Language::Cpp14:
142  sim_cmd = (boost::format("sim_c -S -P Main.cpp / %s/%d/0/* > sim.txt") % accepted_solutions_dir % this->p_id).str();
143  break;
144  case Language::Java:
145  sim_cmd = (boost::format("sim_java -S -P Main.java / %s/%d/2/* > sim.txt") % accepted_solutions_dir % this->p_id).str();
146  break;
147  }
148 
149  int sim_cmd_return_value = system(sim_cmd.c_str());
150  if (-1 == sim_cmd_return_value) {
151  throw JobHandleException("sim execute failed");
152  }
153  if (!WIFEXITED(sim_cmd_return_value) || WEXITSTATUS(sim_cmd_return_value)) {
154  throw JobHandleException("sim command exits incorrectly, exit status: " + std::to_string(WEXITSTATUS(sim_cmd_return_value)));
155  }
156 
157  std::ifstream fp("sim.txt", std::ios::in);
158 
159  if (!fp) {
160  throw JobHandleException("sim.txt open failed");
161  }
162 
163  std::string buf;
164  while (std::getline(fp, buf)) {
165  if (buf.empty()) {
166  break;
167  }
168  }
169 
170  int max_similarity = 0;
171  while (std::getline(fp, buf)) {
172  std::string::reverse_iterator en = std::find(buf.rbegin(), buf.rend(), '%');
173  if (en != buf.rend()) {
174  do {
175  ++en;
176  } while (std::isspace(*en));
177  std::string::reverse_iterator be = std::find_if_not(en, buf.rend(), ::isdigit);
178  --en;
179  --be;
180  try {
181  int now_similarity = std::stoi(buf.substr(buf.length() - (std::distance(buf.rbegin(), be)) - 1, std::distance(en, be)));
182  max_similarity = std::max(max_similarity, now_similarity);
183  } catch (const std::invalid_argument & e) {
184  throw JobHandleException("an error occurred while calculate max similarity");
185  }
186  }
187  }
188 
189  return max_similarity;
190 }

406 Result JudgeJob::execute(const Config & config) const noexcept
407 {
408  using namespace std::chrono;
409  using namespace kerbal::compatibility::chrono_suffix;
410 
411  Result result;
412 
413  // check whether current user is root
414  if (getuid() != 0) {
415  LOG_FATAL(jobType, s_id, log_fp, "ROOT_REQUIRED");
416  result.setErrorCode(RunnerError::ROOT_REQUIRED);
417  return result;
418  }
419 
420  // check args
421  if (config.check_is_valid_config() == false) {
422  LOG_FATAL(jobType, s_id, log_fp, "INVALID_CONFIG");
423  result.setErrorCode(RunnerError::INVALID_CONFIG);
424  return result;
425  }
426 
427  process child_process;
428 
429  // 此处分离了一个运行子进程，稍后该进程内自我完成工作配置，然后替换为用户编写的程序，用于实际的编译/运行
430  try {
431  child_process = process(std::bind(&JudgeJob::child_process, this, std::cref(config))); // equals to: this->child_process(config);
432  } catch (const std::exception & e) {
433  LOG_FATAL(jobType, s_id, log_fp, "Fork failed. Error information: ", e.what());
434  result.setErrorCode(RunnerError::FORK_FAILED);
435  return result;
436  }
437 
438  // record current time
439  time_point<high_resolution_clock> start(high_resolution_clock::now());
440 
441  if (config.limitRealTime()) {
442  std::thread timeout_killer_thread;
443  try {
444  timeout_killer_thread = std::thread([&child_process](milliseconds timeout) {
445  std::this_thread::sleep_for(timeout + 1_s);
446  if (child_process.kill(SIGKILL) != 0) {
447  //TODO log
448  }
449  }, config.max_real_time);
450  } catch (const std::system_error & e) {
451  child_process.kill(SIGKILL);
452  EXCEPT_FATAL(jobType, s_id, log_fp, "Thread construct failed.", e, " , error code: ", e.code().value());
453  result.setErrorCode(RunnerError::PTHREAD_FAILED);
454  return result;
455  } catch (...) {
456  child_process.kill(SIGKILL);
457  UNKNOWN_EXCEPT_FATAL(jobType, s_id, log_fp, "Thread construct failed.");
458  result.setErrorCode(RunnerError::PTHREAD_FAILED);
459  return result;
460  }
461 
462  try {
463  timeout_killer_thread.detach();
464  } catch (const std::system_error & e) {
465  child_process.kill(SIGKILL);
466  EXCEPT_FATAL(jobType, s_id, log_fp, "Thread detach failed.", e, " , error code: ", e.code().value());
467  result.setErrorCode(RunnerError::PTHREAD_FAILED);
468  return result;
469  } catch (...) {
470  child_process.kill(SIGKILL);
471  UNKNOWN_EXCEPT_FATAL(jobType, s_id, log_fp, "Thread detach failed.");
472  result.setErrorCode(RunnerError::PTHREAD_FAILED);
473  return result;
474  }
475  LOG_DEBUG(jobType, s_id, log_fp, "Timeout thread work success");
476  }
477 
478  int status;
479  struct rusage resource_usage;
480 
481  // wait for child process to terminate
482  // on success, returns the process ID of the child whose state has changed;
483  // On error, -1 is returned.
484  if (child_process.join(&status, WSTOPPED, &resource_usage) == -1) {
485  child_process.kill(SIGKILL);
486  LOG_FATAL(jobType, s_id, log_fp, "WAIT_FAILED");
487  result.setErrorCode(RunnerError::WAIT_FAILED);
488  return result;
489  }
490 
491  result.real_time = duration_cast<milliseconds>(high_resolution_clock::now() - start);
492  result.cpu_time = timevalToChrono(resource_usage.ru_utime) + timevalToChrono(resource_usage.ru_stime);
493  result.memory = kerbal::utility::KB(resource_usage.ru_maxrss);
494  result.exit_code = WEXITSTATUS(status);
495 
496  if (result.exit_code != 0) {
497  result.judge_result = UnitedJudgeResult::RUNTIME_ERROR;
498  return result;
499  }
500  // if signaled
501  if (WIFSIGNALED(status) != 0) {
502  result.signal = WTERMSIG(status);
503  LOG_DEBUG(jobType, s_id, log_fp, "signal: ", result.signal);
504  switch (result.signal) {
505  case SIGSEGV:
506  if (config.limitMemory() && result.memory > config.max_memory) {
507  result.judge_result = UnitedJudgeResult::MEMORY_LIMIT_EXCEEDED;
508  } else {
509  result.judge_result = UnitedJudgeResult::RUNTIME_ERROR;
510  }
511  break;
512  case SIGUSR1:
513  result.judge_result = UnitedJudgeResult::SYSTEM_ERROR;
514  break;
515  default:
516  result.judge_result = UnitedJudgeResult::RUNTIME_ERROR;
517  }
518  } else {
519  if (config.limitMemory() && result.memory > config.max_memory) {
520  result.judge_result = UnitedJudgeResult::MEMORY_LIMIT_EXCEEDED;
521  }
522  }
523  if (config.limitRealTime() && result.real_time > config.max_real_time) {
524  result.judge_result = UnitedJudgeResult::REAL_TIME_LIMIT_EXCEEDED;
525  }
526  if (config.limitCPUTime() && result.cpu_time > config.max_cpu_time) {
527  result.judge_result = UnitedJudgeResult::CPU_TIME_LIMIT_EXCEEDED;
528  }
529  return result;
530 }

533 {

556  result.judge_result = UnitedJudgeResult::COMPILE_ERROR;
557  return result;

561 bool JudgeJob::child_process(const Config & config) const
562 {
563  struct rlimit max_stack;
564  max_stack.rlim_cur = max_stack.rlim_max = static_cast<rlim_t>(config.max_stack.count());
565  if (setrlimit(RLIMIT_STACK, &max_stack) != 0) {

567  raise(SIGUSR1);

741 {
