26 inline void add_mysql_conn()
27 {
28  mysqlpp::Connection * mysql_conn = new mysqlpp::Connection(false);
29  try {
30  mysql_conn->set_option(new mysqlpp::SetCharsetNameOption("utf8"));
31  if (!mysql_conn->connect(mysql_database.c_str(), mysql_hostname.c_str(), mysql_username.c_str(), mysql_passwd.c_str())) {
32  throw std::runtime_error("failed connect");
33  }
34  } catch (...) {
35  delete mysql_conn;
36  mysql_conn = nullptr;
37  throw;
38  }
39  mysql_conn_pool.add(mysql_conn);
40  mysql_conn = nullptr;
41 }

43 inline mysql_conn_pool_type::auto_revert_handle sync_fetch_mysql_conn()
44 {
45  mysql_conn_pool_type::auto_revert_handle mysql_conn_handle = mysql_conn_pool.sync_fetch();
46  while (!*mysql_conn_handle) {
47  mysql_conn_handle.abandon();

50  }
