32 RunningConfig::RunningConfig(const JudgeJob & job) :

34 {

37 
38  static const char c_cpp_exe_path[] = "./Main";
39  static const char java_exe_path[] = "/usr/bin/java";
40 
41  switch (job.lang) {
42  case Language::C:
43  case Language::Cpp:
44  case Language::Cpp14: {
45  this->seccomp_rule_name = Seccomp_rule::c_cpp;
46  this->exe_path = c_cpp_exe_path;
47  this->max_memory = job.memoryLimit;
48  this->max_stack = job.memoryLimit;
49  this->max_cpu_time = job.timeLimit;
50  this->args = {c_cpp_exe_path};
51  break;
52  }
53  case Language::Java: {
54  this->seccomp_rule_name = Seccomp_rule::none;
55  this->exe_path = java_exe_path;
56  this->max_memory = MEMORY_UNLIMITED;
57  this->max_stack = 256_MB;
58  this->max_cpu_time = job.timeLimit * 2;
59  static boost::format java_xms("-Xms%dm");
60  static boost::format java_xmx("-Xmx%dm");
61  this->args = {
62  java_exe_path,
63  (java_xms % job.memoryLimit.count()).str(),
64  (java_xmx % (job.memoryLimit.count() + java_memory_bonus.count())).str(),
65  "-Djava.security.manager",
66  "-Djava.security.policy=" + java_policy_path,
67  "Main"
68  };
69  break;
70  }
71  }
72 
73  this->max_real_time = this->max_cpu_time + 1_s;
74  this->max_process_number = UNLIMITED; // TODO
75  this->max_output_size = 32_MB;
76  this->input_path = NULL;
77  this->output_path = "./user.out";
78  this->error_path = "./err.out";

81 CompileConfig::CompileConfig(const JudgeJob & job) :
82  Config()
83 {
84  using namespace kerbal::utility;
85  using namespace kerbal::compatibility::chrono_suffix;
86 
87  // https://icpc.baylor.edu/worldfinals/programming-environment
88  // 2018 ACM ICPC Programming Environment:
89  // gcc -g -O2 -std=gnu11 -static $* -lm
90  // g++ -g -O2 -std=gnu++14 -static $*
91  // javac -encoding UTF-8 -sourcepath . -d . $*
92  // 20180320 add "-Werror=main" in compile_c to avoid "void main()"
93 
94  static const char c_compiler[] = "/usr/bin/gcc";
95  static const char cpp_compiler[] = "/usr/bin/g++";
96  static const char java_compiler[] = "/usr/bin/javac";
97 
98  switch (job.lang) {
99  case Language::C: {
100  this->exe_path = c_compiler;
101  this->args = {c_compiler, "-g", "-O2", "-Werror=main", "-std=gnu11", "-static", "Main.c", "-lm", "-o", "Main"};
102  this->max_memory = 1024_MB;
103  break;
104  }
105  case Language::Cpp: {
106  this->exe_path = cpp_compiler;
107  this->args = {cpp_compiler, "-g", "-O2", "-std=gnu++11", "-static", "Main.cpp", "-o", "Main"};
108  this->max_memory = 1024_MB;
109  break;
110  }
111  case Language::Cpp14: {
112  this->exe_path = cpp_compiler;
113  this->args = {cpp_compiler, "-g", "-O2", "-std=gnu++14", "-static", "Main.cpp", "-o", "Main"};
114  this->max_memory = 1024_MB;
115  break;
116  }
117  case Language::Java: {
118  static boost::format java_xms("-J-Xms%dm");
119  static boost::format java_xmx("-J-Xmx%dm");
120  this->exe_path = java_compiler;
121  this->args = {java_compiler,(java_xms%64).str() , (java_xmx%512).str(), "-encoding", "UTF-8", "-sourcepath", ".", "-d", ".", "Main.java"};
122  this->max_memory = Config::MEMORY_UNLIMITED;
123  break;
124  }
125  }
126  this->max_cpu_time = 5_s;
127  this->max_real_time = 5_s;
128  this->max_stack = 1024_MB;
129  this->max_process_number = Config::UNLIMITED; //TODO
130  this->max_output_size = 128_MB;
131  this->input_path = "/dev/null";
132  this->output_path = "./compiler.out";
133  this->error_path = "./compiler.out";
134  this->seccomp_rule_name = Seccomp_rule::none;
135 }

137 bool Config::check_is_valid_config() const
138 {
139  using namespace kerbal::utility;
140  using namespace std::chrono;
141  using namespace kerbal::compatibility::chrono_suffix;
142 
143  if ((max_cpu_time < 1_ms && max_cpu_time != TIME_UNLIMITED) ||
144 
145  (max_real_time < 1_ms && max_real_time != TIME_UNLIMITED) ||
146 
147  (max_stack < 1_MB) || (max_memory < 1_MB && max_memory != MEMORY_UNLIMITED) ||
148 
149  (max_process_number < 1 && max_process_number != UNLIMITED) ||
150 
151  (max_output_size < 1_MB && max_output_size != MEMORY_UNLIMITED)) {
152  return false;
153  }
154  return true;
155 }

157 RunnerError Config::c_cpp_seccomp_rules() const
158 {
159  // load seccomp rules
160  // 若使用越权的系统调用之后，则强行终止程序
161  using scmp_filter_ptr = std::unique_ptr<void, void (*)(scmp_filter_ctx)>;
162  scmp_filter_ptr ctx(seccomp_init(SCMP_ACT_KILL), seccomp_release);
163  if (ctx == nullptr) {
164  return RunnerError::LOAD_SECCOMP_FAILED;
165  }
166  // 允许的系统调用列表
167  constexpr int syscalls_whitelist[] = { SCMP_SYS(read),
168  SCMP_SYS(fstat),
169  SCMP_SYS(mmap),
170  SCMP_SYS(mprotect),
171  SCMP_SYS(munmap),
172  SCMP_SYS(uname),
173  SCMP_SYS(arch_prctl),
174  SCMP_SYS(brk),
175  SCMP_SYS(access),
176  SCMP_SYS(exit_group),
177  SCMP_SYS(close),
178  SCMP_SYS(readlink),
179  SCMP_SYS(sysinfo),
180  SCMP_SYS(write),
181  SCMP_SYS(writev),
182  SCMP_SYS(lseek)
183  };
184  for (int syscall : syscalls_whitelist) {
185  if (seccomp_rule_add(ctx.get(), SCMP_ACT_ALLOW, syscall, 0) != 0) {
186  return RunnerError::LOAD_SECCOMP_FAILED;
187  }
188  }
189  // add extra rule for execve
190  if (seccomp_rule_add(ctx.get(), SCMP_ACT_ALLOW, SCMP_SYS(execve), 1, SCMP_A0(SCMP_CMP_EQ, (scmp_datum_t )(exe_path))) != 0) {
191  return RunnerError::LOAD_SECCOMP_FAILED;
192  }
193  // do not allow "w" and "rw"
194  if (seccomp_rule_add(ctx.get(), SCMP_ACT_ALLOW, SCMP_SYS(open), 1, SCMP_CMP(1, SCMP_CMP_MASKED_EQ, O_WRONLY | O_RDWR, 0)) != 0) {
195  return RunnerError::LOAD_SECCOMP_FAILED;
196  }
197  // 限制生效
198  if (seccomp_load(ctx.get()) != 0) {
199  return RunnerError::LOAD_SECCOMP_FAILED;
200  }
201  return RunnerError::SUCCESS;
202 }

204 RunnerError Config::general_seccomp_rules() const
205 {
206  // load seccomp rules
207  using scmp_filter_ptr = std::unique_ptr<void, void (*)(scmp_filter_ctx)>;
208  scmp_filter_ptr ctx(seccomp_init(SCMP_ACT_ALLOW), seccomp_release);
209  if (ctx == nullptr) {
210  return RunnerError::LOAD_SECCOMP_FAILED;
211  }
212  // 禁止的系统调用列表
213  constexpr int syscalls_blacklist[] = { SCMP_SYS(socket),
214  SCMP_SYS(clone),
215  SCMP_SYS(fork),
216  SCMP_SYS(vfork),
217  SCMP_SYS(kill)
218  };
219  for (int syscall : syscalls_blacklist) {
220  if (seccomp_rule_add(ctx.get(), SCMP_ACT_KILL, syscall, 0) != 0) {
221  return RunnerError::LOAD_SECCOMP_FAILED;
222  }
223  }
224  // add extra rule for execve
225  if (seccomp_rule_add(ctx.get(), SCMP_ACT_KILL, SCMP_SYS(execve), 1, SCMP_A0(SCMP_CMP_NE, (scmp_datum_t )(exe_path))) != 0) {
226  return RunnerError::LOAD_SECCOMP_FAILED;
227  }
228  // do not allow "w" and "rw"
229  if (seccomp_rule_add(ctx.get(), SCMP_ACT_KILL, SCMP_SYS(open), 1, SCMP_CMP(1, SCMP_CMP_MASKED_EQ, O_WRONLY, O_WRONLY)) != 0) {
230  return RunnerError::LOAD_SECCOMP_FAILED;
231  }
232  if (seccomp_rule_add(ctx.get(), SCMP_ACT_KILL, SCMP_SYS(open), 1, SCMP_CMP(1, SCMP_CMP_MASKED_EQ, O_RDWR, O_RDWR)) != 0) {
233  return RunnerError::LOAD_SECCOMP_FAILED;
234  }
235  // 限制生效
236  if (seccomp_load(ctx.get()) != 0) {
237  return RunnerError::LOAD_SECCOMP_FAILED;
238  }
239  return RunnerError::SUCCESS;
240 }
