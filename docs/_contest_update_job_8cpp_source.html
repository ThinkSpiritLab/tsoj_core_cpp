114 user_problem_status ContestUpdateJob::get_contest_user_problem_status()
115 {
116  LOG_DEBUG(jobType, sid, log_fp, "ContestUpdateJob::get_contest_user_problem_status");
117 
118  mysqlpp::Query query = mysqlConn->query(
119  "select is_ac from contest_user_problem%0 where u_id = %1 and p_id = %2"
120  );
121  query.parse();
122 
123  mysqlpp::StoreQueryResult res = query.store(jobType, uid, pid);
124 
125  if (res.empty()) {
126  if (query.errnum() != 0) {
127  MysqlEmptyResException e(query.errnum(), query.error());
128  EXCEPT_FATAL(jobType, sid, log_fp, "Query user's problem status failed!", e);
129  throw e;
130  }
131  /*
132  * 如果查到的长度为 0, 而且查询过程也没错误, 那么就是 contest_user_problem 里没有这个数据,
133  * 亦即该用户从没提交过此题, 状态为 TODO
134  */
135  return user_problem_status::TODO;
136  }
137 
138  // 注意这里 is_ac 字段 0 1 的含义与练习中的 status 字段是反的
139  switch((int) res[0][0]) {
140  case 0: // is_ac 字段为 0 表示尝试过,
141  return user_problem_status::ATTEMPTED;
142  case 1: // is_ac 字段为 1 表示已 AC
143  return user_problem_status::ACCEPTED;
144  default:
145  LOG_FATAL(jobType, sid, log_fp, "Undefined user's problem status!");
146  throw std::logic_error("Undefined user's problem status!");
147  }
148 }

175 void ContestUpdateJob::update_user_problem()
176 {
177  LOG_DEBUG(jobType, sid, log_fp, "ContestUpdateJob::update_user_problem");
178 
179  bool is_ac = this->result.judge_result == UnitedJudgeResult::ACCEPTED ? true : false;
180  user_problem_status user_problem_status = user_problem_status::TODO;
181  try {
182  user_problem_status = this->get_contest_user_problem_status();
183  } catch (const std::exception & e) {
184  EXCEPT_FATAL(jobType, sid, log_fp, "Get user problem status failed!", e);
185  throw;
186  }
187 
188  bool is_first_ac = false;
189  if (is_ac == true) {
190  try {
191  is_first_ac = this->this_problem_has_not_accepted() == true ? true : false;
192  } catch (const std::exception & e) {
193  EXCEPT_FATAL(jobType, sid, log_fp, "Query is first ac failed!", e);
194  throw;
195  }
196  }
197 
198  switch (user_problem_status) {
199  case user_problem_status::TODO: {
200  int error_count = 0;
201  if (is_ac == false) {
202  error_count++;
203  }
204  mysqlpp::Query insert = mysqlConn->query(
205  "insert into contest_user_problem%0 (u_id, p_id, is_ac, ac_time, error_count, is_first_ac) "
206  "values (%1, %2, %3, %4q, %5, %6)"
207  );
208  insert.parse();
209 
210  mysqlpp::SimpleResult res = insert.execute(jobType, uid, pid, is_ac, postTime, error_count, is_first_ac);
211  if (!res) {
212  MysqlEmptyResException e(insert.errnum(), insert.error());
213  EXCEPT_FATAL(jobType, sid, log_fp, "insert user problem status failed!", e);
214  throw e;
215  }
216  break;
217  }
218  case user_problem_status::ATTEMPTED: {
219  int error_count = this->get_error_count();
220 
221  if (is_ac == false) {
222  error_count++;
223  }
224 
225  mysqlpp::Query update = mysqlConn->query(
226  "update contest_user_problem%0 set is_ac = %1, ac_time = %2q, error_count = %3, is_first_ac = %4 "
227  "where u_id = %5 and p_id = %6"
228  );
229  update.parse();
230 
231  mysqlpp::SimpleResult res = update.execute(jobType, is_ac, postTime, error_count, is_first_ac, uid, pid);
232  if (!res) {
233  MysqlEmptyResException e(update.errnum(), update.error());
234  EXCEPT_FATAL(jobType, sid, log_fp, "update user problem status failed!", e);
235  throw e;
236  }
237  break;
238  }
239  case user_problem_status::ACCEPTED:
240  return;
241  }
242 
243  // update first_ac_user
244  if (is_ac == true && is_first_ac == true) {
245  mysqlpp::Query update = mysqlConn->query(
246  "update contest_problem set first_ac_user = %0 where ct_id = %1 and p_id = %2"
247  );
248  update.parse();
249  mysqlpp::SimpleResult res = update.execute(uid, jobType, pid);
250 
251  if (!res) {
252  MysqlEmptyResException e(update.errnum(), update.error());
253  EXCEPT_FATAL(jobType, sid, log_fp, "update first_ac_user failed!", e);
254  throw e;
255  }
256  }
257  return;
258 }
