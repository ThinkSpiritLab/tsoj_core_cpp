231  static std::string make_exception_description(DupFileType file_type, const char * file_path)
232  {
233  constexpr const char * s[] = { "input file", "output file", "error file" };
234  int i = 0;
235  switch (file_type) {
236  case DupFileType::INPUT_FILE:
237  i = 0;
238  break;
239  case DupFileType::OUTPUT_FILE:
240  i = 1;
241  break;
242  case DupFileType::ERROR_FILE:
243  i = 2;
244  break;
245  }
246  using namespace std::string_literals;
247  return "Dup "s + s[i] + " to [" + file_path + "] failed";
248  }

251  const DupFileType file_type;
252  const std::string file_path;
